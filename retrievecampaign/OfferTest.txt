import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class OfferTest {

    private Offer offer;

    @BeforeEach
    public void setUp() {
        offer = new Offer();
    }

    @Test
    public void testSetAndGetOfferId() {
        offer.setOfferId("12345");
        assertEquals("12345", offer.getOfferId());
    }

    @Test
    public void testSetAndGetFeatureGroup() {
        FeatureGroup__2 featureGroup = new FeatureGroup__2();
        offer.setFeatureGroup(featureGroup);
        assertEquals(featureGroup, offer.getFeatureGroup());
    }

    @Test
    public void testToString() {
        offer.setOfferId("12345");
        FeatureGroup__2 featureGroup = new FeatureGroup__2();
        offer.setFeatureGroup(featureGroup);

        String expected = "com.td.api.mbofp.response.retrievecampaign.Offer@" 
                + Integer.toHexString(System.identityHashCode(offer)) 
                + "[offerId=12345,featureGroup=" + featureGroup + "]";
        
        assertEquals(expected, offer.toString());
    }

    @Test
    public void testHashCode() {
        offer.setOfferId("12345");
        FeatureGroup__2 featureGroup = new FeatureGroup__2();
        offer.setFeatureGroup(featureGroup);

        int expectedHashCode = 31 * (31 + (offer.getOfferId() == null ? 0 : offer.getOfferId().hashCode())) 
                + (offer.getFeatureGroup() == null ? 0 : offer.getFeatureGroup().hashCode());

        assertEquals(expectedHashCode, offer.hashCode());
    }

    @Test
    public void testEquals() {
        offer.setOfferId("12345");
        FeatureGroup__2 featureGroup = new FeatureGroup__2();
        offer.setFeatureGroup(featureGroup);

        Offer anotherOffer = new Offer();
        anotherOffer.setOfferId("12345");
        anotherOffer.setFeatureGroup(featureGroup);

        assertTrue(offer.equals(anotherOffer));

        anotherOffer.setOfferId("67890");
        assertFalse(offer.equals(anotherOffer));

        anotherOffer.setOfferId("12345");
        anotherOffer.setFeatureGroup(new FeatureGroup__2());
        assertFalse(offer.equals(anotherOffer));
    }
}
