import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class NameTest {

    private Name name;

    @BeforeEach
    public void setUp() {
        name = new Name();
    }

    @Test
    public void testSetAndGetText() {
        name.setText("Sample Text");
        assertEquals("Sample Text", name.getText());
    }

    @Test
    public void testSetAndGetLanguageCd() {
        name.setLanguageCd("EN");
        assertEquals("EN", name.getLanguageCd());
    }

    @Test
    public void testToString() {
        name.setText("Sample Text");
        name.setLanguageCd("EN");

        String expected = "com.td.api.mbofp.response.retrievecampaign.Name@" 
                + Integer.toHexString(System.identityHashCode(name)) 
                + "[text=Sample Text,languageCd=EN]";
        
        assertEquals(expected, name.toString());
    }

    @Test
    public void testHashCode() {
        name.setText("Sample Text");
        name.setLanguageCd("EN");

        int expectedHashCode = 31 * (31 + (name.getLanguageCd() == null ? 0 : name.getLanguageCd().hashCode())) 
                + (name.getText() == null ? 0 : name.getText().hashCode());

        assertEquals(expectedHashCode, name.hashCode());
    }

    @Test
    public void testEquals() {
        name.setText("Sample Text");
        name.setLanguageCd("EN");

        Name anotherName = new Name();
        anotherName.setText("Sample Text");
        anotherName.setLanguageCd("EN");

        assertTrue(name.equals(anotherName));

        anotherName.setText("Different Text");
        assertFalse(name.equals(anotherName));

        anotherName.setText("Sample Text");
        anotherName.setLanguageCd("FR");
        assertFalse(name.equals(anotherName));
    }
}
