package com.td.mbna.msec.midtier.auditlog.payload.response;

import org.junit.jupiter.api.Test;

import java.util.Arrays;
import java.util.List;

import static org.junit.jupiter.api.Assertions.*;

class BaseMsecResponseTest {

    @Test
    void testGetAndSetStatus() {
        // Arrange
        BaseMsecResponseImpl response = new BaseMsecResponseImpl();
        String status = "SUCCESS";

        // Act
        response.setStatus(status);

        // Assert
        assertEquals(status, response.getStatus());
    }

    @Test
    void testGetAndSetErrorMsgs() {
        // Arrange
        BaseMsecResponseImpl response = new BaseMsecResponseImpl();
        List<String> errorMsgs = Arrays.asList("Error1", "Error2");

        // Act
        response.setErrorMsgs(errorMsgs);

        // Assert
        assertEquals(errorMsgs, response.getErrorMsgs());
    }

    @Test
    void testGetAndSetResponsePayload() {
        // Arrange
        BaseMsecResponseImpl response = new BaseMsecResponseImpl();
        ResponsePayload payload = new ResponsePayload();
        payload.setBtAmt(1000);

        // Act
        response.setResponsePayload(payload);

        // Assert
        assertEquals(payload, response.getResponsePayload());
    }

    @Test
    void testNullValues() {
        // Arrange
        BaseMsecResponseImpl response = new BaseMsecResponseImpl();

        // Act & Assert
        assertNull(response.getStatus());
        assertNull(response.getErrorMsgs());
        assertNull(response.getResponsePayload());
    }

    @Test
    void testSetEmptyErrorMsgs() {
        // Arrange
        BaseMsecResponseImpl response = new BaseMsecResponseImpl();
        List<String> errorMsgs = Arrays.asList();

        // Act
        response.setErrorMsgs(errorMsgs);

        // Assert
        assertNotNull(response.getErrorMsgs());
        assertTrue(response.getErrorMsgs().isEmpty());
    }
}
