package com.td.mbna.msec.midtier.auditlog.payload.response;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class CreateAuditLogResponseTest {

    @Test
    void testGetAndSetResponsePayload() {
        // Arrange
        CreateAuditLogResponse response = new CreateAuditLogResponse();
        ResponsePayload responsePayload = new ResponsePayload();
        responsePayload.setBtAmt(1000);

        // Act
        response.setResponsePayload(responsePayload);

        // Assert
        assertEquals(responsePayload, response.getResponsePayload());
    }

    @Test
    void testNullResponsePayload() {
        // Arrange
        CreateAuditLogResponse response = new CreateAuditLogResponse();

        // Act & Assert
        assertNull(response.getResponsePayload());
    }

    @Test
    void testSerializationUID() {
        // Act & Assert
        assertEquals(-3255706629309040935L, CreateAuditLogResponse.serialVersionUID);
    }
}
