package com.td.api.mbcca.response.account;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import java.util.ArrayList;
import java.util.List;

import static org.junit.jupiter.api.Assertions.*;

class AgreementLoyaltyInfoTest {

    private AgreementLoyaltyInfo agreementLoyaltyInfo;

    @BeforeEach
    void setUp() {
        agreementLoyaltyInfo = new AgreementLoyaltyInfo();
    }

    @Test
    void testGetEnrolledInd() {
        agreementLoyaltyInfo.setEnrolledInd(true);
        assertTrue(agreementLoyaltyInfo.getEnrolledInd());

        agreementLoyaltyInfo.setEnrolledInd(false);
        assertFalse(agreementLoyaltyInfo.getEnrolledInd());
    }

    @Test
    void testSetEnrolledInd() {
        agreementLoyaltyInfo.setEnrolledInd(true);
        assertTrue(agreementLoyaltyInfo.getEnrolledInd());

        agreementLoyaltyInfo.setEnrolledInd(false);
        assertFalse(agreementLoyaltyInfo.getEnrolledInd());
    }

    @Test
    void testGetAccountRewardQualifierInfo() {
        List<AccountRewardQualifierInfo> rewardQualifierInfoList = new ArrayList<>();
        AccountRewardQualifierInfo info = new AccountRewardQualifierInfo();
        rewardQualifierInfoList.add(info);

        agreementLoyaltyInfo.setAccountRewardQualifierInfo(rewardQualifierInfoList);

        assertEquals(rewardQualifierInfoList, agreementLoyaltyInfo.getAccountRewardQualifierInfo());
    }

    @Test
    void testSetAccountRewardQualifierInfo() {
        List<AccountRewardQualifierInfo> rewardQualifierInfoList = new ArrayList<>();
        AccountRewardQualifierInfo info = new AccountRewardQualifierInfo();
        rewardQualifierInfoList.add(info);

        agreementLoyaltyInfo.setAccountRewardQualifierInfo(rewardQualifierInfoList);
        assertEquals(rewardQualifierInfoList, agreementLoyaltyInfo.getAccountRewardQualifierInfo());
    }

    @Test
    void testToString() {
        agreementLoyaltyInfo.setEnrolledInd(true);
        assertNotNull(agreementLoyaltyInfo.toString());
    }

    @Test
    void testHashCode() {
        AgreementLoyaltyInfo agreementLoyaltyInfo2 = new AgreementLoyaltyInfo();
        agreementLoyaltyInfo.setEnrolledInd(true);
        agreementLoyaltyInfo2.setEnrolledInd(true);

        assertEquals(agreementLoyaltyInfo.hashCode(), agreementLoyaltyInfo2.hashCode());
    }

    @Test
    void testEquals() {
        AgreementLoyaltyInfo agreementLoyaltyInfo2 = new AgreementLoyaltyInfo();
        agreementLoyaltyInfo.setEnrolledInd(true);
        agreementLoyaltyInfo2.setEnrolledInd(true);

        assertTrue(agreementLoyaltyInfo.equals(agreementLoyaltyInfo2));

        agreementLoyaltyInfo2.setEnrolledInd(false);
        assertFalse(agreementLoyaltyInfo.equals(agreementLoyaltyInfo2));
    }
}
