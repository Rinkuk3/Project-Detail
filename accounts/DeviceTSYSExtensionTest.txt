import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class DeviceTSYSExtensionTest {

    private DeviceTSYSExtension deviceTSYSExtension;

    @BeforeEach
    void setUp() {
        deviceTSYSExtension = new DeviceTSYSExtension();
    }

    @Test
    void testGetAndSetAccountNum() {
        deviceTSYSExtension.setAccountNum("123456");
        assertEquals("123456", deviceTSYSExtension.getAccountNum());
    }

    @Test
    void testToString() {
        deviceTSYSExtension.setAccountNum("123456");
        
        String expectedString = "com.td.api.mbcca.response.account.DeviceTSYSExtension@" + 
                Integer.toHexString(System.identityHashCode(deviceTSYSExtension)) + 
                "[accountNum=123456]";
        
        assertEquals(expectedString, deviceTSYSExtension.toString());
    }

    @Test
    void testHashCode() {
        deviceTSYSExtension.setAccountNum("123456");
        
        int expectedHashCode = 1;
        expectedHashCode = ((expectedHashCode * 31) + ("123456" == null ? 0 : "123456".hashCode()));

        assertEquals(expectedHashCode, deviceTSYSExtension.hashCode());
    }

    @Test
    void testEquals() {
        DeviceTSYSExtension anotherDeviceTSYSExtension = new DeviceTSYSExtension();
        deviceTSYSExtension.setAccountNum("123456");

        anotherDeviceTSYSExtension.setAccountNum("123456");

        assertTrue(deviceTSYSExtension.equals(anotherDeviceTSYSExtension));
        
        anotherDeviceTSYSExtension.setAccountNum("654321");
        assertFalse(deviceTSYSExtension.equals(anotherDeviceTSYSExtension));
    }
}
