import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class ElectricTransactionInfoTest {

    private ElectricTransactionInfo electricTransactionInfo;

    @BeforeEach
    void setUp() {
        electricTransactionInfo = new ElectricTransactionInfo();
    }

    @Test
    void testGetAndSetElectronicCommerceIndicatorCd() {
        electricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        assertEquals("EC123", electricTransactionInfo.getElectronicCommerceIndicatorCd());
    }

    @Test
    void testGetAndSetSecurityLevelIndicatorCd() {
        electricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        assertEquals("SL123", electricTransactionInfo.getSecurityLevelIndicatorCd());
    }

    @Test
    void testGetAndSetSecurityProtocolCd() {
        electricTransactionInfo.setSecurityProtocolCd("SP123");
        assertEquals("SP123", electricTransactionInfo.getSecurityProtocolCd());
    }

    @Test
    void testGetAndSetCardHolderAuthenticationCd() {
        electricTransactionInfo.setCardHolderAuthenticationCd("CH123");
        assertEquals("CH123", electricTransactionInfo.getCardHolderAuthenticationCd());
    }

    @Test
    void testToString() {
        electricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        electricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        electricTransactionInfo.setSecurityProtocolCd("SP123");
        electricTransactionInfo.setCardHolderAuthenticationCd("CH123");

        String expectedString = "com.td.api.mbcca.response.account.ElectricTransactionInfo@" + 
                Integer.toHexString(System.identityHashCode(electricTransactionInfo)) + 
                "[electronicCommerceIndicatorCd=EC123,securityLevelIndicatorCd=SL123,securityProtocolCd=SP123,cardHolderAuthenticationCd=CH123]";
        
        assertEquals(expectedString, electricTransactionInfo.toString());
    }

    @Test
    void testHashCode() {
        electricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        electricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        electricTransactionInfo.setSecurityProtocolCd("SP123");
        electricTransactionInfo.setCardHolderAuthenticationCd("CH123");

        ElectricTransactionInfo anotherElectricTransactionInfo = new ElectricTransactionInfo();
        anotherElectricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        anotherElectricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        anotherElectricTransactionInfo.setSecurityProtocolCd("SP123");
        anotherElectricTransactionInfo.setCardHolderAuthenticationCd("CH123");

        assertEquals(electricTransactionInfo.hashCode(), anotherElectricTransactionInfo.hashCode());
    }

    @Test
    void testEquals() {
        electricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        electricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        electricTransactionInfo.setSecurityProtocolCd("SP123");
        electricTransactionInfo.setCardHolderAuthenticationCd("CH123");

        ElectricTransactionInfo anotherElectricTransactionInfo = new ElectricTransactionInfo();
        anotherElectricTransactionInfo.setElectronicCommerceIndicatorCd("EC123");
        anotherElectricTransactionInfo.setSecurityLevelIndicatorCd("SL123");
        anotherElectricTransactionInfo.setSecurityProtocolCd("SP123");
        anotherElectricTransactionInfo.setCardHolderAuthenticationCd("CH123");

        assertTrue(electricTransactionInfo.equals(anotherElectricTransactionInfo));

        anotherElectricTransactionInfo.setElectronicCommerceIndicatorCd("EC456");
        assertFalse(electricTransactionInfo.equals(anotherElectricTransactionInfo));
    }
}
