package com.td.api.mbcca.response.account;

import com.td.api.mbcca.shared.Cccyamt;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class BillingInfoTest {

    private BillingInfo billingInfo;
    private Cccyamt balanceCccyamt;
    private Cccyamt financeChargeCccyamt;

    @BeforeEach
    public void setUp() {
        billingInfo = new BillingInfo();
        balanceCccyamt = new Cccyamt(); // Mock or create a new Cccyamt object
        financeChargeCccyamt = new Cccyamt(); // Mock or create a new Cccyamt object
    }

    @Test
    public void testGetSetAprRate() {
        Double aprRate = 12.5;
        billingInfo.setAprRate(aprRate);
        assertEquals(aprRate, billingInfo.getAprRate());
    }

    @Test
    public void testGetSetAprTimePeriodCd() {
        String aprTimePeriodCd = "MONTHLY";
        billingInfo.setAprTimePeriodCd(aprTimePeriodCd);
        assertEquals(aprTimePeriodCd, billingInfo.getAprTimePeriodCd());
    }

    @Test
    public void testGetSetPeriodicRate() {
        Double periodicRate = 1.5;
        billingInfo.setPeriodicRate(periodicRate);
        assertEquals(periodicRate, billingInfo.getPeriodicRate());
    }

    @Test
    public void testGetSetBalanceCccyamt() {
        billingInfo.setBalanceCccyamt(balanceCccyamt);
        assertEquals(balanceCccyamt, billingInfo.getBalanceCccyamt());
    }

    @Test
    public void testGetSetFinanceChargeCccyamt() {
        billingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);
        assertEquals(financeChargeCccyamt, billingInfo.getFinanceChargeCccyamt());
    }

    @Test
    public void testToString() {
        billingInfo.setAprRate(12.5);
        billingInfo.setAprTimePeriodCd("MONTHLY");
        billingInfo.setPeriodicRate(1.5);
        billingInfo.setBalanceCccyamt(balanceCccyamt);
        billingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        String expectedString = BillingInfo.class.getName() + "@" + Integer.toHexString(System.identityHashCode(billingInfo)) + "[" +
                "aprRate=" + "12.5" + ',' +
                "aprTimePeriodCd=" + "MONTHLY" + ',' +
                "periodicRate=" + "1.5" + ',' +
                "balanceCccyamt=" + balanceCccyamt + ',' +
                "financeChargeCccyamt=" + financeChargeCccyamt + ']';

        assertEquals(expectedString, billingInfo.toString());
    }

    @Test
    public void testHashCode() {
        billingInfo.setAprRate(12.5);
        billingInfo.setAprTimePeriodCd("MONTHLY");
        billingInfo.setPeriodicRate(1.5);
        billingInfo.setBalanceCccyamt(balanceCccyamt);
        billingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        assertEquals(billingInfo.hashCode(), billingInfo.hashCode());
    }

    @Test
    public void testEquals() {
        BillingInfo anotherBillingInfo = new BillingInfo();
        anotherBillingInfo.setAprRate(12.5);
        anotherBillingInfo.setAprTimePeriodCd("MONTHLY");
        anotherBillingInfo.setPeriodicRate(1.5);
        anotherBillingInfo.setBalanceCccyamt(balanceCccyamt);
        anotherBillingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        billingInfo.setAprRate(12.5);
        billingInfo.setAprTimePeriodCd("MONTHLY");
        billingInfo.setPeriodicRate(1.5);
        billingInfo.setBalanceCccyamt(balanceCccyamt);
        billingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        assertTrue(billingInfo.equals(anotherBillingInfo));
    }

    @Test
    public void testEqualsWithNull() {
        BillingInfo anotherBillingInfo = null;
        assertFalse(billingInfo.equals(anotherBillingInfo));
    }

    @Test
    public void testEqualsWithDifferentObject() {
        String someString = "Different Object";
        assertFalse(billingInfo.equals(someString));
    }

    @Test
    public void testEqualsWithDifferentValues() {
        BillingInfo anotherBillingInfo = new BillingInfo();
        anotherBillingInfo.setAprRate(15.0); // Different value
        anotherBillingInfo.setAprTimePeriodCd("YEARLY");
        anotherBillingInfo.setPeriodicRate(2.0);
        anotherBillingInfo.setBalanceCccyamt(balanceCccyamt);
        anotherBillingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        billingInfo.setAprRate(12.5);
        billingInfo.setAprTimePeriodCd("MONTHLY");
        billingInfo.setPeriodicRate(1.5);
        billingInfo.setBalanceCccyamt(balanceCccyamt);
        billingInfo.setFinanceChargeCccyamt(financeChargeCccyamt);

        assertFalse(billingInfo.equals(anotherBillingInfo));
    }
}
