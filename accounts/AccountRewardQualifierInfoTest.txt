import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class AccountRewardQualifierInfoTest {

    @Test
    void testGetSetAccountRewardQualifierCd() {
        AccountRewardQualifierInfo info = new AccountRewardQualifierInfo();
        String value = "TestQualifier";
        
        // Test setter
        info.setAccountRewardQualifierCd(value);
        // Test getter
        assertEquals(value, info.getAccountRewardQualifierCd());
    }

    @Test
    void testToString() {
        AccountRewardQualifierInfo info = new AccountRewardQualifierInfo();
        info.setAccountRewardQualifierCd("TestQualifier");
        
        String result = info.toString();
        assertTrue(result.contains("accountRewardQualifierCd=TestQualifier"));
    }

    @Test
    void testHashCode() {
        AccountRewardQualifierInfo info1 = new AccountRewardQualifierInfo();
        info1.setAccountRewardQualifierCd("TestQualifier");
        
        AccountRewardQualifierInfo info2 = new AccountRewardQualifierInfo();
        info2.setAccountRewardQualifierCd("TestQualifier");
        
        assertEquals(info1.hashCode(), info2.hashCode());
    }

    @Test
    void testEquals() {
        AccountRewardQualifierInfo info1 = new AccountRewardQualifierInfo();
        info1.setAccountRewardQualifierCd("TestQualifier");

        AccountRewardQualifierInfo info2 = new AccountRewardQualifierInfo();
        info2.setAccountRewardQualifierCd("TestQualifier");

        AccountRewardQualifierInfo info3 = new AccountRewardQualifierInfo();
        info3.setAccountRewardQualifierCd("AnotherQualifier");

        assertTrue(info1.equals(info2)); // Equal objects
        assertFalse(info1.equals(info3)); // Different objects
        assertFalse(info1.equals(null)); // Compare with null
        assertFalse(info1.equals(new Object())); // Compare with different class
    }
}
