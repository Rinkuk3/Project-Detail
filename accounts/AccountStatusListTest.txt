import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import java.util.Arrays;

class AccountStatusListTest {

    @Test
    void testGetSetAccountStatus() {
        AccountStatusList statusList = new AccountStatusList();
        
        Accountstatus status1 = new Accountstatus();
        status1.setCd("Active");
        
        Accountstatus status2 = new Accountstatus();
        status2.setCd("Inactive");
        
        statusList.setAccountStatus(Arrays.asList(status1, status2));
        
        // Test getter
        assertNotNull(statusList.getAccountStatus());
        assertEquals(2, statusList.getAccountStatus().size());
        assertEquals("Active", statusList.getAccountStatus().get(0).getCd());
        assertEquals("Inactive", statusList.getAccountStatus().get(1).getCd());
    }

    @Test
    void testToString() {
        AccountStatusList statusList = new AccountStatusList();
        
        Accountstatus status1 = new Accountstatus();
        status1.setCd("Active");
        
        Accountstatus status2 = new Accountstatus();
        status2.setCd("Inactive");
        
        statusList.setAccountStatus(Arrays.asList(status1, status2));
        
        String result = statusList.toString();
        assertTrue(result.contains("accountStatus"));
    }
}
