package com.td.api.mbcca.response.account;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class AgreementToProductTest {

    private AgreementToProduct agreementToProduct;

    @BeforeEach
    void setUp() {
        agreementToProduct = new AgreementToProduct();
    }

    @Test
    void testGetSecuritizationPoolId() {
        agreementToProduct.setSecuritizationPoolId("poolId123");
        assertEquals("poolId123", agreementToProduct.getSecuritizationPoolId());
    }

    @Test
    void testSetSecuritizationPoolId() {
        agreementToProduct.setSecuritizationPoolId("poolId123");
        assertEquals("poolId123", agreementToProduct.getSecuritizationPoolId());
    }

    @Test
    void testToString() {
        agreementToProduct.setSecuritizationPoolId("poolId123");
        assertNotNull(agreementToProduct.toString());
    }

    @Test
    void testHashCode() {
        AgreementToProduct agreementToProduct2 = new AgreementToProduct();
        agreementToProduct.setSecuritizationPoolId("poolId123");
        agreementToProduct2.setSecuritizationPoolId("poolId123");
        assertEquals(agreementToProduct.hashCode(), agreementToProduct2.hashCode());
    }

    @Test
    void testEquals() {
        AgreementToProduct agreementToProduct2 = new AgreementToProduct();
        agreementToProduct.setSecuritizationPoolId("poolId123");
        agreementToProduct2.setSecuritizationPoolId("poolId123");

        assertTrue(agreementToProduct.equals(agreementToProduct2));

        agreementToProduct2.setSecuritizationPoolId("poolId456");
        assertFalse(agreementToProduct.equals(agreementToProduct2));
    }
}
