package com.td.api.mbcca.response.account;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class TsysExtensionTest {

    @Test
    void testSettersAndGetters() {
        TsysExtension tsysExtension = new TsysExtension();

        tsysExtension.setCustomerId("12345");
        tsysExtension.setBusinessContactName("John Doe");
        tsysExtension.setPowerOfAttorneyName("Jane Smith");
        tsysExtension.setLastName("Doe");
        tsysExtension.setFirstName("John");
        tsysExtension.setMiddleName("Michael");
        tsysExtension.setTitlePrefix("Mr.");
        tsysExtension.setBirthDt("1990-01-01");
        tsysExtension.setIdentityVerificationVal("Verified");
        tsysExtension.setCardNum("1111222233334444");
        tsysExtension.setCardNumMasked("****222233334444");
        tsysExtension.setNameSuffix("Jr.");

        assertEquals("12345", tsysExtension.getCustomerId());
        assertEquals("John Doe", tsysExtension.getBusinessContactName());
        assertEquals("Jane Smith", tsysExtension.getPowerOfAttorneyName());
        assertEquals("Doe", tsysExtension.getLastName());
        assertEquals("John", tsysExtension.getFirstName());
        assertEquals("Michael", tsysExtension.getMiddleName());
        assertEquals("Mr.", tsysExtension.getTitlePrefix());
        assertEquals("1990-01-01", tsysExtension.getBirthDt());
        assertEquals("Verified", tsysExtension.getIdentityVerificationVal());
        assertEquals("1111222233334444", tsysExtension.getCardNum());
        assertEquals("****222233334444", tsysExtension.getCardNumMasked());
        assertEquals("Jr.", tsysExtension.getNameSuffix());
    }

    @Test
    void testToString() {
        TsysExtension tsysExtension = new TsysExtension();
        tsysExtension.setCustomerId("12345");
        tsysExtension.setBusinessContactName("John Doe");

        String expectedToString = "com.td.api.mbcca.response.account.TsysExtension@" + Integer.toHexString(System.identityHashCode(tsysExtension)) +
                "[customerId=12345,businessContactName=John Doe,powerOfAttorneyName=<null>,lastName=<null>,firstName=<null>,middleName=<null>,titlePrefix=<null>,birthDt=<null>,identityVerificationVal=<null>,cardNum=<null>,cardNumMasked=<null>,nameSuffix=<null>]";

        assertTrue(tsysExtension.toString().startsWith("com.td.api.mbcca.response.account.TsysExtension@"));
        assertTrue(tsysExtension.toString().contains("customerId=12345"));
        assertTrue(tsysExtension.toString().contains("businessContactName=John Doe"));
    }

    @Test
    void testHashCode() {
        TsysExtension tsysExtension1 = new TsysExtension();
        tsysExtension1.setCustomerId("12345");
        tsysExtension1.setBusinessContactName("John Doe");

        TsysExtension tsysExtension2 = new TsysExtension();
        tsysExtension2.setCustomerId("12345");
        tsysExtension2.setBusinessContactName("John Doe");

        TsysExtension tsysExtension3 = new TsysExtension();
        tsysExtension3.setCustomerId("54321");
        tsysExtension3.setBusinessContactName("Jane Doe");

        assertEquals(tsysExtension1.hashCode(), tsysExtension2.hashCode());
        assertNotEquals(tsysExtension1.hashCode(), tsysExtension3.hashCode());
    }

    @Test
    void testEquals() {
        TsysExtension tsysExtension1 = new TsysExtension();
        tsysExtension1.setCustomerId("12345");
        tsysExtension1.setBusinessContactName("John Doe");

        TsysExtension tsysExtension2 = new TsysExtension();
        tsysExtension2.setCustomerId("12345");
        tsysExtension2.setBusinessContactName("John Doe");

        TsysExtension tsysExtension3 = new TsysExtension();
        tsysExtension3.setCustomerId("54321");
        tsysExtension3.setBusinessContactName("Jane Doe");

        assertTrue(tsysExtension1.equals(tsysExtension2));
        assertFalse(tsysExtension1.equals(tsysExtension3));
        assertFalse(tsysExtension1.equals(null));
        assertFalse(tsysExtension1.equals(new Object()));
    }

    @Test
    void testEqualsWithNullValues() {
        TsysExtension tsysExtension1 = new TsysExtension();
        tsysExtension1.setCustomerId(null);
        tsysExtension1.setBusinessContactName(null);

        TsysExtension tsysExtension2 = new TsysExtension();
        tsysExtension2.setCustomerId(null);
        tsysExtension2.setBusinessContactName(null);

        TsysExtension tsysExtension3 = new TsysExtension();
        tsysExtension3.setCustomerId("12345");
        tsysExtension3.setBusinessContactName("John Doe");

        assertTrue(tsysExtension1.equals(tsysExtension2));
        assertFalse(tsysExtension1.equals(tsysExtension3));
    }
}
