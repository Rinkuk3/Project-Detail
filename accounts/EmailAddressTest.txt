import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;
import java.util.Date;

class EmailAddressTest {

    private EmailAddress emailAddress;

    @BeforeEach
    void setUp() {
        emailAddress = new EmailAddress();
    }

    @Test
    void testGetAndSetStartDt() {
        emailAddress.setStartDt("2023-03-01");
        assertEquals("2023-03-01", emailAddress.getStartDt());
    }

    @Test
    void testGetAndSetEndDt() {
        emailAddress.setEndDt("2023-03-31");
        assertEquals("2023-03-31", emailAddress.getEndDt());
    }

    @Test
    void testGetAndSetStartDttm() {
        Date startDate = new Date();
        emailAddress.setStartDttm(startDate);
        assertEquals(startDate, emailAddress.getStartDttm());
    }

    @Test
    void testGetAndSetEndDttm() {
        Date endDate = new Date();
        emailAddress.setEndDttm(endDate);
        assertEquals(endDate, emailAddress.getEndDttm());
    }

    @Test
    void testGetAndSetText() {
        emailAddress.setText("test@example.com");
        assertEquals("test@example.com", emailAddress.getText());
    }

    @Test
    void testGetAndSetDomainName() {
        emailAddress.setDomainName("example.com");
        assertEquals("example.com", emailAddress.getDomainName());
    }

    @Test
    void testGetAndSetDesc() {
        emailAddress.setDesc("Email Address Description");
        assertEquals("Email Address Description", emailAddress.getDesc());
    }

    @Test
    void testGetAndSetLastUpdateDttm() {
        Date lastUpdate = new Date();
        emailAddress.setLastUpdateDttm(lastUpdate);
        assertEquals(lastUpdate, emailAddress.getLastUpdateDttm());
    }

    @Test
    void testToString() {
        emailAddress.setStartDt("2023-03-01");
        emailAddress.setEndDt("2023-03-31");
        emailAddress.setStartDttm(new Date());
        emailAddress.setEndDttm(new Date());
        emailAddress.setText("test@example.com");
        emailAddress.setDomainName("example.com");
        emailAddress.setDesc("Email Address Description");
        emailAddress.setLastUpdateDttm(new Date());

        String expectedString = "com.td.api.mbcca.response.account.EmailAddress@" + 
                Integer.toHexString(System.identityHashCode(emailAddress)) + 
                "[startDt=2023-03-01,endDt=2023-03-31,startDttm=" + emailAddress.getStartDttm() + 
                ",endDttm=" + emailAddress.getEndDttm() + ",text=test@example.com,domainName=example.com,desc=Email Address Description,lastUpdateDttm=" + emailAddress.getLastUpdateDttm() + "]";
        
        assertEquals(expectedString, emailAddress.toString());
    }

    @Test
    void testHashCode() {
        emailAddress.setStartDt("2023-03-01");
        emailAddress.setEndDt("2023-03-31");
        emailAddress.setStartDttm(new Date());
        emailAddress.setEndDttm(new Date());
        emailAddress.setText("test@example.com");
        emailAddress.setDomainName("example.com");
        emailAddress.setDesc("Email Address Description");
        emailAddress.setLastUpdateDttm(new Date());

        EmailAddress anotherEmailAddress = new EmailAddress();
        anotherEmailAddress.setStartDt("2023-03-01");
        anotherEmailAddress.setEndDt("2023-03-31");
        anotherEmailAddress.setStartDttm(new Date());
        anotherEmailAddress.setEndDttm(new Date());
        anotherEmailAddress.setText("test@example.com");
        anotherEmailAddress.setDomainName("example.com");
        anotherEmailAddress.setDesc("Email Address Description");
        anotherEmailAddress.setLastUpdateDttm(new Date());

        assertEquals(emailAddress.hashCode(), anotherEmailAddress.hashCode());
    }

    @Test
    void testEquals() {
        emailAddress.setStartDt("2023-03-01");
        emailAddress.setEndDt("2023-03-31");
        emailAddress.setStartDttm(new Date());
        emailAddress.setEndDttm(new Date());
        emailAddress.setText("test@example.com");
        emailAddress.setDomainName("example.com");
        emailAddress.setDesc("Email Address Description");
        emailAddress.setLastUpdateDttm(new Date());

        EmailAddress anotherEmailAddress = new EmailAddress();
        anotherEmailAddress.setStartDt("2023-03-01");
        anotherEmailAddress.setEndDt("2023-03-31");
        anotherEmailAddress.setStartDttm(new Date());
        anotherEmailAddress.setEndDttm(new Date());
        anotherEmailAddress.setText("test@example.com");
        anotherEmailAddress.setDomainName("example.com");
        anotherEmailAddress.setDesc("Email Address Description");
        anotherEmailAddress.setLastUpdateDttm(new Date());

        assertTrue(emailAddress.equals(anotherEmailAddress));

        anotherEmailAddress.setStartDt("2023-04-01");
        assertFalse(emailAddress.equals(anotherEmailAddress));
    }
}
