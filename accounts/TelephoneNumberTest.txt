package com.td.api.mbcca.response.account;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import java.util.Date;

import static org.junit.jupiter.api.Assertions.*;

class TelephoneNumberTest {

    private TelephoneNumber telephoneNumber;

    @BeforeEach
    void setUp() {
        telephoneNumber = new TelephoneNumber();
    }

    @Test
    void testGetAndSetStartDt() {
        String startDt = "2025-01-01";
        telephoneNumber.setStartDt(startDt);
        assertEquals(startDt, telephoneNumber.getStartDt());
    }

    @Test
    void testGetAndSetEndDt() {
        String endDt = "2025-12-31";
        telephoneNumber.setEndDt(endDt);
        assertEquals(endDt, telephoneNumber.getEndDt());
    }

    @Test
    void testGetAndSetStartDttm() {
        Date startDttm = new Date();
        telephoneNumber.setStartDttm(startDttm);
        assertEquals(startDttm, telephoneNumber.getStartDttm());
    }

    @Test
    void testGetAndSetEndDttm() {
        Date endDttm = new Date();
        telephoneNumber.setEndDttm(endDttm);
        assertEquals(endDttm, telephoneNumber.getEndDttm());
    }

    @Test
    void testGetAndSetTelephoneNum() {
        String telephoneNum = "1234567890";
        telephoneNumber.setTelephoneNum(telephoneNum);
        assertEquals(telephoneNum, telephoneNumber.getTelephoneNum());
    }

    @Test
    void testGetAndSetCountryCodeNum() {
        String countryCodeNum = "1";
        telephoneNumber.setCountryCodeNum(countryCodeNum);
        assertEquals(countryCodeNum, telephoneNumber.getCountryCodeNum());
    }

    @Test
    void testGetAndSetAreaCodeNum() {
        String areaCodeNum = "415";
        telephoneNumber.setAreaCodeNum(areaCodeNum);
        assertEquals(areaCodeNum, telephoneNumber.getAreaCodeNum());
    }

    @Test
    void testGetAndSetExchangeNum() {
        String exchangeNum = "555";
        telephoneNumber.setExchangeNum(exchangeNum);
        assertEquals(exchangeNum, telephoneNumber.getExchangeNum());
    }

    @Test
    void testGetAndSetLineNum() {
        String lineNum = "1234";
        telephoneNumber.setLineNum(lineNum);
        assertEquals(lineNum, telephoneNumber.getLineNum());
    }

    @Test
    void testGetAndSetExtensionNum() {
        String extensionNum = "567";
        telephoneNumber.setExtensionNum(extensionNum);
        assertEquals(extensionNum, telephoneNumber.getExtensionNum());
    }

    @Test
    void testGetAndSetLastUpdateDttm() {
        Date lastUpdateDttm = new Date();
        telephoneNumber.setLastUpdateDttm(lastUpdateDttm);
        assertEquals(lastUpdateDttm, telephoneNumber.getLastUpdateDttm());
    }

    @Test
    void testToString() {
        String expectedString = "com.td.api.mbcca.response.account.TelephoneNumber@" + Integer.toHexString(telephoneNumber.hashCode()) + "[startDt=<null>,endDt=<null>,startDttm=<null>,endDttm=<null>,telephoneNum=<null>,countryCodeNum=<null>,areaCodeNum=<null>,exchangeNum=<null>,lineNum=<null>,extensionNum=<null>,lastUpdateDttm=<null>]";
        assertTrue(telephoneNumber.toString().contains("TelephoneNumber"));
    }

    @Test
    void testHashCode() {
        int initialHashCode = telephoneNumber.hashCode();
        assertNotNull(initialHashCode);
    }

    @Test
    void testEquals() {
        TelephoneNumber telephoneNumber1 = new TelephoneNumber();
        telephoneNumber1.setTelephoneNum("1234567890");
        telephoneNumber1.setCountryCodeNum("1");

        TelephoneNumber telephoneNumber2 = new TelephoneNumber();
        telephoneNumber2.setTelephoneNum("1234567890");
        telephoneNumber2.setCountryCodeNum("1");

        assertTrue(telephoneNumber1.equals(telephoneNumber2));

        telephoneNumber2.setTelephoneNum("0987654321");
        assertFalse(telephoneNumber1.equals(telephoneNumber2));
    }

    @Test
    void testNotEqualsDifferentObject() {
        assertFalse(telephoneNumber.equals(new Object()));
    }
}
