import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class PartyToTransferEventTest {

    private PartyToTransferEvent partyToTransferEvent;

    @BeforeEach
    void setUp() {
        partyToTransferEvent = new PartyToTransferEvent();
    }

    @Test
    void testGetAndSetEventStartDt() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        assertEquals("2025-03-03", partyToTransferEvent.getEventStartDt());
    }

    @Test
    void testGetAndSetSourceAccountNum() {
        partyToTransferEvent.setSourceAccountNum("123456789");
        assertEquals("123456789", partyToTransferEvent.getSourceAccountNum());
    }

    @Test
    void testGetAndSetDestinationAccountNum() {
        partyToTransferEvent.setDestinationAccountNum("987654321");
        assertEquals("987654321", partyToTransferEvent.getDestinationAccountNum());
    }

    @Test
    void testToString() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        String expectedToString = "com.td.api.mbcca.response.account.PartyToTransferEvent@" +
                Integer.toHexString(System.identityHashCode(partyToTransferEvent)) +
                "[eventStartDt=2025-03-03,sourceAccountNum=123456789,destinationAccountNum=987654321,]";

        assertEquals(expectedToString, partyToTransferEvent.toString());
    }

    @Test
    void testHashCode() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertEquals(partyToTransferEvent.hashCode(), anotherPartyToTransferEvent.hashCode());
    }

    @Test
    void testEquals() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertTrue(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithDifferentSourceAccountNum() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum("111111111");
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertFalse(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithDifferentDestinationAccountNum() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum("222222222");

        assertFalse(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithDifferentEventStartDt() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-04-04");
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertFalse(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithNullEventStartDt() {
        partyToTransferEvent.setEventStartDt(null);
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt(null);
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertTrue(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithNullSourceAccountNum() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum(null);
        partyToTransferEvent.setDestinationAccountNum("987654321");

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum(null);
        anotherPartyToTransferEvent.setDestinationAccountNum("987654321");

        assertTrue(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }

    @Test
    void testEqualsWithNullDestinationAccountNum() {
        partyToTransferEvent.setEventStartDt("2025-03-03");
        partyToTransferEvent.setSourceAccountNum("123456789");
        partyToTransferEvent.setDestinationAccountNum(null);

        PartyToTransferEvent anotherPartyToTransferEvent = new PartyToTransferEvent();
        anotherPartyToTransferEvent.setEventStartDt("2025-03-03");
        anotherPartyToTransferEvent.setSourceAccountNum("123456789");
        anotherPartyToTransferEvent.setDestinationAccountNum(null);

        assertTrue(partyToTransferEvent.equals(anotherPartyToTransferEvent));
    }
}
