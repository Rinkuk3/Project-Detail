import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class MerchantInfoTest {

    @Test
    public void testSetGetMerchantNum() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setMerchantNum("12345");
        assertEquals("12345", merchantInfo.getMerchantNum());
    }

    @Test
    public void testSetGetMerchantCategoryCd() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setMerchantCategoryCd("MCC001");
        assertEquals("MCC001", merchantInfo.getMerchantCategoryCd());
    }

    @Test
    public void testSetGetName() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setName("Test Merchant");
        assertEquals("Test Merchant", merchantInfo.getName());
    }

    @Test
    public void testSetGetCity() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setCity("New York");
        assertEquals("New York", merchantInfo.getCity());
    }

    @Test
    public void testSetGetTerritory() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setTerritory("NY");
        assertEquals("NY", merchantInfo.getTerritory());
    }

    @Test
    public void testSetGetCountry() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setCountry("USA");
        assertEquals("USA", merchantInfo.getCountry());
    }

    @Test
    public void testSetGetPostalCode() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setPostalCode("10001");
        assertEquals("10001", merchantInfo.getPostalCode());
    }

    @Test
    public void testSetGetTelephoneNum() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setTelephoneNum("123-456-7890");
        assertEquals("123-456-7890", merchantInfo.getTelephoneNum());
    }

    @Test
    public void testSetGetMerchantCategoryCodeDesc() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setMerchantCategoryCodeDesc("Category Description");
        assertEquals("Category Description", merchantInfo.getMerchantCategoryCodeDesc());
    }

    @Test
    public void testToString() {
        MerchantInfo merchantInfo = new MerchantInfo();
        merchantInfo.setMerchantNum("12345");
        merchantInfo.setMerchantCategoryCd("MCC001");
        merchantInfo.setName("Test Merchant");
        merchantInfo.setCity("New York");
        merchantInfo.setTerritory("NY");
        merchantInfo.setCountry("USA");
        merchantInfo.setPostalCode("10001");
        merchantInfo.setTelephoneNum("123-456-7890");
        merchantInfo.setMerchantCategoryCodeDesc("Category Description");

        String expectedString = "com.td.api.mbcca.shared.MerchantInfo@" + Integer.toHexString(merchantInfo.hashCode()) + "[merchantNum=12345,merchantCategoryCd=MCC001,name=Test Merchant,city=New York,territory=NY,country=USA,postalCode=10001,telephoneNum=123-456-7890,merchantCategoryCodeDesc=Category Description,]";

        assertEquals(expectedString, merchantInfo.toString());
    }

    @Test
    public void testHashCode() {
        MerchantInfo merchantInfo1 = new MerchantInfo();
        merchantInfo1.setMerchantNum("12345");
        merchantInfo1.setMerchantCategoryCd("MCC001");

        MerchantInfo merchantInfo2 = new MerchantInfo();
        merchantInfo2.setMerchantNum("12345");
        merchantInfo2.setMerchantCategoryCd("MCC001");

        assertEquals(merchantInfo1.hashCode(), merchantInfo2.hashCode());
    }

    @Test
    public void testEquals() {
        MerchantInfo merchantInfo1 = new MerchantInfo();
        merchantInfo1.setMerchantNum("12345");
        merchantInfo1.setMerchantCategoryCd("MCC001");

        MerchantInfo merchantInfo2 = new MerchantInfo();
        merchantInfo2.setMerchantNum("12345");
        merchantInfo2.setMerchantCategoryCd("MCC001");

        assertTrue(merchantInfo1.equals(merchantInfo2));
    }

    @Test
    public void testNotEquals() {
        MerchantInfo merchantInfo1 = new MerchantInfo();
        merchantInfo1.setMerchantNum("12345");
        merchantInfo1.setMerchantCategoryCd("MCC001");

        MerchantInfo merchantInfo2 = new MerchantInfo();
        merchantInfo2.setMerchantNum("12346");
        merchantInfo2.setMerchantCategoryCd("MCC002");

        assertFalse(merchantInfo1.equals(merchantInfo2));
    }
}
