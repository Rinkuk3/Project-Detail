import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class TdCardsExtensionEventDescriptionTest {

    @Test
    public void testSetGetEventDescriptionTxt() {
        TdCardsExtensionEventDescription eventDescription = new TdCardsExtensionEventDescription();
        eventDescription.setEventDescriptionTxt("Event Description");
        assertEquals("Event Description", eventDescription.getEventDescriptionTxt());
    }

    @Test
    public void testToString() {
        TdCardsExtensionEventDescription eventDescription = new TdCardsExtensionEventDescription();
        eventDescription.setEventDescriptionTxt("Event Description");
        String expectedString = "com.td.api.mbcca.shared.TdCardsExtensionEventDescription@" + Integer.toHexString(eventDescription.hashCode()) + "[eventDescriptionTxt=Event Description,]";
        assertEquals(expectedString, eventDescription.toString());
    }

    @Test
    public void testHashCode() {
        TdCardsExtensionEventDescription eventDescription1 = new TdCardsExtensionEventDescription();
        eventDescription1.setEventDescriptionTxt("Event Description");

        TdCardsExtensionEventDescription eventDescription2 = new TdCardsExtensionEventDescription();
        eventDescription2.setEventDescriptionTxt("Event Description");

        assertEquals(eventDescription1.hashCode(), eventDescription2.hashCode());
    }

    @Test
    public void testEquals() {
        TdCardsExtensionEventDescription eventDescription1 = new TdCardsExtensionEventDescription();
        eventDescription1.setEventDescriptionTxt("Event Description");

        TdCardsExtensionEventDescription eventDescription2 = new TdCardsExtensionEventDescription();
        eventDescription2.setEventDescriptionTxt("Event Description");

        assertTrue(eventDescription1.equals(eventDescription2));
    }

    @Test
    public void testNotEquals() {
        TdCardsExtensionEventDescription eventDescription1 = new TdCardsExtensionEventDescription();
        eventDescription1.setEventDescriptionTxt("Event Description 1");

        TdCardsExtensionEventDescription eventDescription2 = new TdCardsExtensionEventDescription();
        eventDescription2.setEventDescriptionTxt("Event Description 2");

        assertFalse(eventDescription1.equals(eventDescription2));
    }

    @Test
    public void testEqualsWithNullEventDescriptionTxt() {
        TdCardsExtensionEventDescription eventDescription1 = new TdCardsExtensionEventDescription();
        eventDescription1.setEventDescriptionTxt(null);

        TdCardsExtensionEventDescription eventDescription2 = new TdCardsExtensionEventDescription();
        eventDescription2.setEventDescriptionTxt(null);

        assertTrue(eventDescription1.equals(eventDescription2));
    }

    @Test
    public void testEqualsWithNull() {
        TdCardsExtensionEventDescription eventDescription = new TdCardsExtensionEventDescription();
        eventDescription.setEventDescriptionTxt("Event Description");

        assertFalse(eventDescription.equals(null));
    }

    @Test
    public void testEqualsWithDifferentClass() {
        TdCardsExtensionEventDescription eventDescription = new TdCardsExtensionEventDescription();
        eventDescription.setEventDescriptionTxt("Event Description");

        String notEventDescription = "Not a TdCardsExtensionEventDescription";

        assertFalse(eventDescription.equals(notEventDescription));
    }

}
