import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class TsysExtensionCurrencyNumericCdTest {

    @Test
    public void testSetGetCurrencyNumericCd() {
        TsysExtensionCurrencyNumericCd currency = new TsysExtensionCurrencyNumericCd();
        currency.setCurrencyNumericCd("840");
        assertEquals("840", currency.getCurrencyNumericCd());
    }

    @Test
    public void testToString() {
        TsysExtensionCurrencyNumericCd currency = new TsysExtensionCurrencyNumericCd();
        currency.setCurrencyNumericCd("840");
        String expectedString = "com.td.api.mbcca.shared.TsysExtensionCurrencyNumericCd@" + Integer.toHexString(currency.hashCode()) + "[currencyNumericCd=840,]";
        assertEquals(expectedString, currency.toString());
    }

    @Test

        TsysExtensionCurrencyNumericCd currency1 = new TsysExtensionCurrencyNumericCd();
        currency1.setCurrencyNumericCd("840");

        TsysExtensionCurrencyNumericCd currency2 = new TsysExtensionCurrencyNumericCd();
        currency2.setCurrencyNumericCd("840");

        assertEquals(currency1.hashCode(), currency2.hashCode());
    }

    @Test
    public void testEquals() {
        TsysExtensionCurrencyNumericCd currency1 = new TsysExtensionCurrencyNumericCd();
        currency1.setCurrencyNumericCd("840");

        TsysExtensionCurrencyNumericCd currency2 = new TsysExtensionCurrencyNumericCd();
        currency2.setCurrencyNumericCd("840");

        assertTrue(currency1.equals(currency2));
    }

    @Test
    public void testNotEquals() {
        TsysExtensionCurrencyNumericCd currency1 = new TsysExtensionCurrencyNumericCd();
        currency1.setCurrencyNumericCd("840");

        TsysExtensionCurrencyNumericCd currency2 = new TsysExtensionCurrencyNumericCd();
        currency2.setCurrencyNumericCd("978");

        assertFalse(currency1.equals(currency2));
    }

    @Test
    public void testEqualsWithNullCurrencyNumericCd() {
        TsysExtensionCurrencyNumericCd currency1 = new TsysExtensionCurrencyNumericCd();
        currency1.setCurrencyNumericCd(null);

        TsysExtensionCurrencyNumericCd currency2 = new TsysExtensionCurrencyNumericCd();
        currency2.setCurrencyNumericCd(null);

        assertTrue(currency1.equals(currency2));
    }

    @Test
    public void testEqualsWithNull() {
        TsysExtensionCurrencyNumericCd currency = new TsysExtensionCurrencyNumericCd();
        currency.setCurrencyNumericCd("840");

        assertFalse(currency.equals(null));
    }

    @Test
    public void testEqualsWithDifferentClass() {
        TsysExtensionCurrencyNumericCd currency = new TsysExtensionCurrencyNumericCd();
        currency.setCurrencyNumericCd("840");

        String notCurrency = "Not a TsysExtensionCurrencyNumericCd";

        assertFalse(currency.equals(notCurrency));
