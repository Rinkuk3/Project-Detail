import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class PartyResponseTest {

    private PartyResponse partyResponse;

    @BeforeEach
    public void setUp() {
        partyResponse = new PartyResponse();
    }

    @Test
    public void testSetAndGetEventId() {
        String eventId = "12345";
        partyResponse.setEventId(eventId);
        assertEquals(eventId, partyResponse.getEventId());
    }

    @Test
    public void testToString() {
        partyResponse.setEventId("12345");

        String expected = "com.td.api.mbofp.response.updateofferdisposition.PartyResponse@"
                + Integer.toHexString(System.identityHashCode(partyResponse)) + "[eventId=12345,]";
        assertEquals(expected, partyResponse.toString());
    }

    @Test
    public void testHashCode() {
        partyResponse.setEventId("12345");

        int expectedHashCode = 1;
        expectedHashCode = (expectedHashCode * 31) + ("12345" == null ? 0 : "12345".hashCode());

        assertEquals(expectedHashCode, partyResponse.hashCode());
    }

    @Test
    public void testEquals() {
        partyResponse.setEventId("12345");

        PartyResponse anotherPartyResponse = new PartyResponse();
        anotherPartyResponse.setEventId("12345");

        assertTrue(partyResponse.equals(anotherPartyResponse));

        anotherPartyResponse.setEventId("67890");
        assertFalse(partyResponse.equals(anotherPartyResponse));

        assertFalse(partyResponse.equals(null));
        assertFalse(partyResponse.equals(new Object()));
    }
}
