import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class CreditCardProductPackageTest {

    private CreditCardProductPackage creditCardProductPackage;
    private FeatureGroup__1 featureGroup;
    private CreditCardProduct creditCardProduct;

    @BeforeEach
    public void setUp() {
        creditCardProductPackage = new CreditCardProductPackage();
        
        featureGroup = new FeatureGroup__1();
        creditCardProduct = new CreditCardProduct();
    }

    @Test
    public void testSetAndGetProductPackageNum() {
        String productPackageNum = "12345";
        creditCardProductPackage.setProductPackageNum(productPackageNum);
        assertEquals(productPackageNum, creditCardProductPackage.getProductPackageNum());
    }

    @Test
    public void testSetAndGetProductPackageDesc() {
        String productPackageDesc = "Credit Card Package Description";
        creditCardProductPackage.setProductPackageDesc(productPackageDesc);
        assertEquals(productPackageDesc, creditCardProductPackage.getProductPackageDesc());
    }

    @Test
    public void testSetAndGetFeatureGroup() {
        creditCardProductPackage.setFeatureGroup(featureGroup);
        assertEquals(featureGroup, creditCardProductPackage.getFeatureGroup());
    }

    @Test
    public void testSetAndGetCreditCardProduct() {
        creditCardProductPackage.setCreditCardProduct(creditCardProduct);
        assertEquals(creditCardProduct, creditCardProductPackage.getCreditCardProduct());
    }

    @Test
    public void testToString() {
        creditCardProductPackage.setProductPackageNum("12345");
        creditCardProductPackage.setProductPackageDesc("Credit Card Package Description");
        creditCardProductPackage.setFeatureGroup(featureGroup);
        creditCardProductPackage.setCreditCardProduct(creditCardProduct);

        String expected = "com.td.api.mbofp.response.retrieveofferlist.CreditCardProductPackage@" 
                + Integer.toHexString(System.identityHashCode(creditCardProductPackage)) 
                + "[productPackageNum=12345,productPackageDesc=Credit Card Package Description,featureGroup=" 
                + featureGroup + ",creditCardProduct=" + creditCardProduct + "]";

        assertEquals(expected, creditCardProductPackage.toString());
    }

    @Test
    public void testHashCode() {
        creditCardProductPackage.setProductPackageNum("12345");
        creditCardProductPackage.setProductPackageDesc("Credit Card Package Description");
        creditCardProductPackage.setFeatureGroup(featureGroup);
        creditCardProductPackage.setCreditCardProduct(creditCardProduct);

        int expectedHashCode = 31 + 
                (creditCardProductPackage.getProductPackageNum() == null ? 0 : creditCardProductPackage.getProductPackageNum().hashCode()) +
                (creditCardProductPackage.getProductPackageDesc() == null ? 0 : creditCardProductPackage.getProductPackageDesc().hashCode()) +
                (creditCardProductPackage.getFeatureGroup() == null ? 0 : creditCardProductPackage.getFeatureGroup().hashCode()) +
                (creditCardProductPackage.getCreditCardProduct() == null ? 0 : creditCardProductPackage.getCreditCardProduct().hashCode());

        assertEquals(expectedHashCode, creditCardProductPackage.hashCode());
    }

    @Test
    public void testEquals() {
        creditCardProductPackage.setProductPackageNum("12345");
        creditCardProductPackage.setProductPackageDesc("Credit Card Package Description");
        creditCardProductPackage.setFeatureGroup(featureGroup);
        creditCardProductPackage.setCreditCardProduct(creditCardProduct);

        CreditCardProductPackage anotherCreditCardProductPackage = new CreditCardProductPackage();
        anotherCreditCardProductPackage.setProductPackageNum("12345");
        anotherCreditCardProductPackage.setProductPackageDesc("Credit Card Package Description");
        anotherCreditCardProductPackage.setFeatureGroup(featureGroup);
        anotherCreditCardProductPackage.setCreditCardProduct(creditCardProduct);

        assertTrue(creditCardProductPackage.equals(anotherCreditCardProductPackage));

        anotherCreditCardProductPackage.setProductPackageNum("54321");
        assertFalse(creditCardProductPackage.equals(anotherCreditCardProductPackage));

        anotherCreditCardProductPackage.setProductPackageNum("12345");
        anotherCreditCardProductPackage.setFeatureGroup(new FeatureGroup__1());
        assertFalse(creditCardProductPackage.equals(anotherCreditCardProductPackage));

        assertFalse(creditCardProductPackage.equals(null));
        assertFalse(creditCardProductPackage.equals(new Object()));
    }
}
