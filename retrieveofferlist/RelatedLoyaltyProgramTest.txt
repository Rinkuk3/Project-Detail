import com.td.api.mbofp.shared.LoyaltyProgram;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class RelatedLoyaltyProgramTest {

    private RelatedLoyaltyProgram relatedLoyaltyProgram;

    @BeforeEach
    public void setUp() {
        relatedLoyaltyProgram = new RelatedLoyaltyProgram();
    }

    @Test
    public void testSetAndGetLoyaltyProgram() {
        LoyaltyProgram loyaltyProgram = new LoyaltyProgram();
        relatedLoyaltyProgram.setLoyaltyProgram(loyaltyProgram);
        assertEquals(loyaltyProgram, relatedLoyaltyProgram.getLoyaltyProgram());
    }

    @Test
    public void testSetAndGetProductRelationshipTypeCd() {
        String productRelationshipTypeCd = "REL_TYPE_CD";
        relatedLoyaltyProgram.setProductRelationshipTypeCd(productRelationshipTypeCd);
        assertEquals(productRelationshipTypeCd, relatedLoyaltyProgram.getProductRelationshipTypeCd());
    }

    @Test
    public void testSetAndGetMembershipRequiredInd() {
        Boolean membershipRequiredInd = true;
        relatedLoyaltyProgram.setMembershipRequiredInd(membershipRequiredInd);
        assertEquals(membershipRequiredInd, relatedLoyaltyProgram.getMembershipRequiredInd());
    }

    @Test
    public void testToString() {
        LoyaltyProgram loyaltyProgram = new LoyaltyProgram();
        String productRelationshipTypeCd = "REL_TYPE_CD";
        Boolean membershipRequiredInd = true;

        relatedLoyaltyProgram.setLoyaltyProgram(loyaltyProgram);
        relatedLoyaltyProgram.setProductRelationshipTypeCd(productRelationshipTypeCd);
        relatedLoyaltyProgram.setMembershipRequiredInd(membershipRequiredInd);

        String expected = "com.td.api.mbofp.response.retrieveofferlist.RelatedLoyaltyProgram@"
                + Integer.toHexString(System.identityHashCode(relatedLoyaltyProgram))
                + "[loyaltyProgram=" + loyaltyProgram
                + ",productRelationshipTypeCd=" + productRelationshipTypeCd
                + ",membershipRequiredInd=" + membershipRequiredInd + "]";
        assertEquals(expected, relatedLoyaltyProgram.toString());
    }

    @Test
    public void testHashCode() {
        LoyaltyProgram loyaltyProgram = new LoyaltyProgram();
        String productRelationshipTypeCd = "REL_TYPE_CD";
        Boolean membershipRequiredInd = true;

        relatedLoyaltyProgram.setLoyaltyProgram(loyaltyProgram);
        relatedLoyaltyProgram.setProductRelationshipTypeCd(productRelationshipTypeCd);
        relatedLoyaltyProgram.setMembershipRequiredInd(membershipRequiredInd);

        int expectedHashCode = 1;
        expectedHashCode = (expectedHashCode * 31) + (relatedLoyaltyProgram.getMembershipRequiredInd() == null ? 0 : relatedLoyaltyProgram.getMembershipRequiredInd().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (relatedLoyaltyProgram.getLoyaltyProgram() == null ? 0 : relatedLoyaltyProgram.getLoyaltyProgram().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (relatedLoyaltyProgram.getProductRelationshipTypeCd() == null ? 0 : relatedLoyaltyProgram.getProductRelationshipTypeCd().hashCode());

        assertEquals(expectedHashCode, relatedLoyaltyProgram.hashCode());
    }

    @Test
    public void testEquals() {
        LoyaltyProgram loyaltyProgram = new LoyaltyProgram();
        String productRelationshipTypeCd = "REL_TYPE_CD";
        Boolean membershipRequiredInd = true;

        relatedLoyaltyProgram.setLoyaltyProgram(loyaltyProgram);
        relatedLoyaltyProgram.setProductRelationshipTypeCd(productRelationshipTypeCd);
        relatedLoyaltyProgram.setMembershipRequiredInd(membershipRequiredInd);

        RelatedLoyaltyProgram anotherRelatedLoyaltyProgram = new RelatedLoyaltyProgram();
        anotherRelatedLoyaltyProgram.setLoyaltyProgram(loyaltyProgram);
        anotherRelatedLoyaltyProgram.setProductRelationshipTypeCd(productRelationshipTypeCd);
        anotherRelatedLoyaltyProgram.setMembershipRequiredInd(membershipRequiredInd);

        assertTrue(relatedLoyaltyProgram.equals(anotherRelatedLoyaltyProgram));

        anotherRelatedLoyaltyProgram.setProductRelationshipTypeCd("DIFFERENT_REL_TYPE_CD");
        assertFalse(relatedLoyaltyProgram.equals(anotherRelatedLoyaltyProgram));

        assertFalse(relatedLoyaltyProgram.equals(null));
        assertFalse(relatedLoyaltyProgram.equals(new Object()));
    }
}
