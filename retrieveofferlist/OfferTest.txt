    @Test

        offer.setOfferDesc(offerDesc);

    }

    @Test
    public void testSetAndGetTypeCd() {

        offer.setTypeCd(typeCd);
        assertEquals(typeCd, offer.getTypeCd());

    @Test

        Promotion promotion = new Promotion();
        offer.setPromotion(promotion);

    @Test

        FeatureGroup__3 featureGroup = new FeatureGroup__3();

    @Test
    public void testToString() {
        Promotion promotion = new Promotion();
        FeatureGroup__3 featureGroup = new FeatureGroup__3();
        String offerId = "OFFER123";
        String offerDesc = "Special Offer";
        String typeCd = "TYPE01";
        String promotionOfferDispositionCd = "DISPOSITION01";

        offer.setOfferId(offerId);
        offer.setOfferDesc(offerDesc);
        offer.setTypeCd(typeCd);
        offer.setPromotion(promotion);
        offer.setFeatureGroup(featureGroup);
        offer.setPromotionOfferDispositionCd(promotionOfferDispositionCd);

        String expected = "com.td.api.mbofp.response.retrieveofferlist.Offer@"
                + Integer.toHexString(System.identityHashCode(offer))
                + "[offerId=" + offerId + ",offerDesc=" + offerDesc + ",typeCd=" + typeCd
                + ",promotion=" + promotion + ",featureGroup=" + featureGroup
                + ",promotionOfferDispositionCd=" + promotionOfferDispositionCd + "]";
        assertEquals(expected, offer.toString());
    }

    @Test
    public void testHashCode() {
        Promotion promotion = new Promotion();
        FeatureGroup__3 featureGroup = new FeatureGroup__3();
        String offerId = "OFFER123";
        String offerDesc = "Special Offer";
        String typeCd = "TYPE01";
        String promotionOfferDispositionCd = "DISPOSITION01";

        offer.setOfferId(offerId);
        offer.setOfferDesc(offerDesc);
        offer.setTypeCd(typeCd);
        offer.setPromotion(promotion);
        offer.setFeatureGroup(featureGroup);
        offer.setPromotionOfferDispositionCd(promotionOfferDispositionCd);

        int expectedHashCode = 1;
        expectedHashCode = (expectedHashCode * 31) + (offer.getOfferDesc() == null ? 0 : offer.getOfferDesc().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (offer.getTypeCd() == null ? 0 : offer.getTypeCd().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (offer.getFeatureGroup() == null ? 0 : offer.getFeatureGroup().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (offer.getOfferId() == null ? 0 : offer.getOfferId().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (offer.getPromotion() == null ? 0 : offer.getPromotion().hashCode());
        expectedHashCode = (expectedHashCode * 31) + (offer.getPromotionOfferDispositionCd() == null ? 0 : offer.getPromotionOfferDispositionCd().hashCode());

        assertEquals(expectedHashCode, offer.hashCode());
    }

    @Test
    public void testEquals() {
        Promotion promotion = new Promotion();
        FeatureGroup__3 featureGroup = new FeatureGroup__3();
        String offerId = "OFFER123";
        String offerDesc = "Special Offer";
        String typeCd = "TYPE01";
        String promotionOfferDispositionCd = "DISPOSITION01";

        offer.setOfferId(offerId);
        offer.setOfferDesc(offerDesc);
        offer.setTypeCd(typeCd);
        offer.setPromotion(promotion);
        offer.setFeatureGroup(featureGroup);
        offer.setPromotionOfferDispositionCd(promotionOfferDispositionCd);

        Offer anotherOffer = new Offer();
        anotherOffer.setOfferId(offerId);
        anotherOffer.setOfferDesc(offerDesc);
        anotherOffer.setTypeCd(typeCd);
        anotherOffer.setPromotion(promotion);
        anotherOffer.setFeatureGroup(featureGroup);
        anotherOffer.setPromotionOfferDispositionCd(promotionOfferDispositionCd);

        assertTrue(offer.equals(anotherOffer));

        anotherOffer.setOfferDesc("Different Offer");
        assertFalse(offer.equals(anotherOffer));

        assertFalse(offer.equals(null));
        assertFalse(offer.equals(new Object()));
    }
}
