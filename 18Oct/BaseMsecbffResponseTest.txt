import static org.assertj.core.api.Assertions.assertThat;

import org.junit.jupiter.api.Test;

public class BaseMsecbffResponseTest {

    @Test
    public void testBuilder() {
        // Given
        String expectedStatus = "success";
        String expectedErrorCode = "200";

        // When
        BaseMsecbffResponse response = BaseMsecbffResponse.builder()
                .status(expectedStatus)
                .errorCode(expectedErrorCode)
                .build();

        // Then
        assertThat(response).isNotNull();
        assertThat(response.getStatus()).isEqualTo(expectedStatus);
        assertThat(response.getErrorCode()).isEqualTo(expectedErrorCode);
    }

    @Test
    public void testGettersAndSetters() {
        // Given
        BaseMsecbffResponse response = new BaseMsecbffResponse();

        // When
        response.setStatus("success");
        response.setErrorCode("200");

        // Then
        assertThat(response.getStatus()).isEqualTo("success");
        assertThat(response.getErrorCode()).isEqualTo("200");
    }

    @Test
    public void testEqualsAndHashCode() {
        // Given
        BaseMsecbffResponse response1 = BaseMsecbffResponse.builder()
                .status("success")
                .errorCode("200")
                .build();
        
        BaseMsecbffResponse response2 = BaseMsecbffResponse.builder()
                .status("success")
                .errorCode("200")
                .build();

        BaseMsecbffResponse response3 = BaseMsecbffResponse.builder()
                .status("error")
                .errorCode("400")
                .build();

        // When & Then
        assertThat(response1).isEqualTo(response2);
        assertThat(response1.hashCode()).isEqualTo(response2.hashCode());
        assertThat(response1).isNotEqualTo(response3);
    }

    @Test
    public void testToString() {
        // Given
        BaseMsecbffResponse response = BaseMsecbffResponse.builder()
                .status("success")
                .errorCode("200")
                .build();

        // When
        String responseString = response.toString();

        // Then
        assertThat(responseString).contains("status='success'")
                                  .contains("errorCode='200'");
    }
}
