import static org.junit.jupiter.api.Assertions.*;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class CustomerNameTest {

    private CustomerName customerName;

    @BeforeEach
    public void setUp() {
        customerName = new CustomerName("John", "Doe");
    }

    @Test
    public void testConstructor() {
        assertEquals("John", customerName.getFirstname());
        assertEquals("Doe", customerName.getLastname());
    }

    @Test
    public void testGetFirstname() {
        assertEquals("John", customerName.getFirstname());
    }

    @Test
    public void testSetFirstname() {
        customerName.setFirstname("Jane");
        assertEquals("Jane", customerName.getFirstname());
    }

    @Test
    public void testGetLastname() {
        assertEquals("Doe", customerName.getLastname());
    }

    @Test
    public void testSetLastname() {
        customerName.setLastname("Smith");
        assertEquals("Smith", customerName.getLastname());
    }
}
