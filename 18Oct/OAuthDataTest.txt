import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class OAuthDataTest {

    @Test
    public void testGettersAndSetters() {
        // Given
        OAuthData oauthData = new OAuthData();
        String expectedAccessToken = "someAccessToken";
        long expectedExpiry = 3600L; // 1 hour in seconds
        String expectedConnectId = "connectId123";
        String expectedToken = "token123";

        // When
        oauthData.setAccessToken(expectedAccessToken);
        oauthData.setExpiry(expectedExpiry);
        oauthData.setConnectId(expectedConnectId);
        oauthData.setToken(expectedToken);

        // Then
        assertThat(oauthData.getAccessToken()).isEqualTo(expectedAccessToken);
        assertThat(oauthData.getExpiry()).isEqualTo(expectedExpiry);
        assertThat(oauthData.getConnectId()).isEqualTo(expectedConnectId);
        assertThat(oauthData.getToken()).isEqualTo(expectedToken);
    }
}
