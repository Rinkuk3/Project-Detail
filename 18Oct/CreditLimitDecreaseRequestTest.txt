package com.td.mbna.msec.creditlimit.model;

import static org.junit.jupiter.api.Assertions.*;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class CreditLimitDecreaseRequestTest {

    private CreditLimitDecreaseRequest creditLimitDecreaseRequest;

    @BeforeEach
    public void setUp() {
        creditLimitDecreaseRequest = new CreditLimitDecreaseRequest();
    }

    @Test
    public void testGetRequestedCreditLimit() {
        // Verify the default value is null
        assertNull(creditLimitDecreaseRequest.getRequestedCreditLimit());

        // Set a value and verify
        String expectedValue = "5000";
        creditLimitDecreaseRequest.setRequestedCreditLimit(expectedValue);
        assertEquals(expectedValue, creditLimitDecreaseRequest.getRequestedCreditLimit());
    }

    @Test
    public void testSetRequestedCreditLimit() {
        String expectedValue = "3000";
        creditLimitDecreaseRequest.setRequestedCreditLimit(expectedValue);

        // Verify that the value was set correctly
        assertEquals(expectedValue, creditLimitDecreaseRequest.getRequestedCreditLimit());
    }
}
