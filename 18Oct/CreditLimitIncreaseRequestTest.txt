package com.td.mbna.msec.creditlimit.model;

import static org.junit.jupiter.api.Assertions.*;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class CreditLimitIncreaseRequestTest {

    private CreditLimitIncreaseRequest request;

    @BeforeEach
    public void setUp() {
        request = new CreditLimitIncreaseRequest();
    }

    @Test
    public void testSetAndGetAccountId() {
        String accountId = "123456";
        request.setAccountId(accountId);
        assertEquals(accountId, request.getAccountId());
    }

    @Test
    public void testSetAndGetRequestedCreditLimit() {
        String requestedCreditLimit = "5000";
        request.setRequestedCreditLimit(requestedCreditLimit);
        assertEquals(requestedCreditLimit, request.getRequestedCreditLimit());
    }

    @Test
    public void testToString() {
        request.setAccountId("123456");
        request.setRequestedCreditLimit("5000");
        String expectedString = "CreditLimitIncreaseRequest{accountId='123456', requestedCreditLimit='5000'}";
        assertEquals(expectedString, request.toString());
    }
}
