import static org.junit.jupiter.api.Assertions.*;

import com.td.mbna.msec.cardlock.model.CardControlRequest;
import org.junit.jupiter.api.Test;

public class CardControlRequestTest {

    @Test
    public void testGetSetCustomerId() {
        // Arrange
        CardControlRequest request = new CardControlRequest();
        String customerId = "12345";

        // Act
        request.setCustomerId(customerId);

        // Assert
        assertEquals(customerId, request.getCustomerId());
    }

    @Test
    public void testGetSetActive() {
        // Arrange
        CardControlRequest request = new CardControlRequest();
        Boolean active = true;

        // Act
        request.setActive(active);

        // Assert
        assertEquals(active, request.getActive());
    }

    @Test
    public void testSetActive_NullValue() {
        // Arrange
        CardControlRequest request = new CardControlRequest();

        // Act
        request.setActive(null);

        // Assert
        assertNull(request.getActive());
    }
}
