import org.junit.jupiter.api.Test;

import java.util.Locale;

import static org.assertj.core.api.Assertions.assertThat;

public class LocaleWrapperTest {

    @Test
    public void testConstructorAndGetLocale() {
        // Given
        Locale expectedLocale = Locale.US;

        // When
        LocaleWrapper localeWrapper = new LocaleWrapper(expectedLocale);

        // Then
        assertThat(localeWrapper.getLocale()).isEqualTo(expectedLocale);
    }

    @Test
    public void testSetLocale() {
        // Given
        Locale initialLocale = Locale.US;
        LocaleWrapper localeWrapper = new LocaleWrapper(initialLocale);
        Locale newLocale = Locale.CANADA;

        // When
        localeWrapper.setLocale(newLocale);

        // Then
        assertThat(localeWrapper.getLocale()).isEqualTo(newLocale);
    }
}
