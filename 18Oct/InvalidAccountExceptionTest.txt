import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class InvalidAccountExceptionTest {

    @Test
    public void testInvalidAccountExceptionMessage() {
        // Given
        String accountNumber = "123456789";
        
        // When
        InvalidAccountException exception = new InvalidAccountException(accountNumber);

        // Then
        assertThat(exception.getMessage())
                .isEqualTo("Invalid account: Account with number " + accountNumber + " does not exist!");
    }

    @Test
    public void testInvalidAccountExceptionSerialization() {
        // Given
        String accountNumber = "987654321";
        InvalidAccountException exception = new InvalidAccountException(accountNumber);

        // When
        String expectedMessage = "Invalid account: Account with number " + accountNumber + " does not exist!";
        
        // Then
        assertThat(exception.getMessage()).isEqualTo(expectedMessage);
        assertThat(exception).isInstanceOf(MBNAException.class);
    }
}
