import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class TokenHolderTest {

    @Test
    public void testDefaultConstructor() {
        // Given
        TokenHolder tokenHolder = new TokenHolder();

        // When
        String jwtToken = tokenHolder.getJwtToken();
        String connectId = tokenHolder.getConnectId();

        // Then
        assertThat(jwtToken).isNull();
        assertThat(connectId).isNull();
    }

    @Test
    public void testParameterizedConstructor() {
        // Given
        String expectedJwtToken = "someJwtToken";
        String expectedConnectId = "connectId123";

        // When
        TokenHolder tokenHolder = new TokenHolder(expectedJwtToken, expectedConnectId);

        // Then
        assertThat(tokenHolder.getJwtToken()).isEqualTo(expectedJwtToken);
        assertThat(tokenHolder.getConnectId()).isEqualTo(expectedConnectId);
    }

    @Test
    public void testSetters() {
        // Given
        TokenHolder tokenHolder = new TokenHolder();

        String expectedJwtToken = "newJwtToken";
        String expectedConnectId = "newConnectId";

        // When
        tokenHolder.setJwtToken(expectedJwtToken);
        tokenHolder.setConnectId(expectedConnectId);

        // Then
        assertThat(tokenHolder.getJwtToken()).isEqualTo(expectedJwtToken);
        assertThat(tokenHolder.getConnectId()).isEqualTo(expectedConnectId);
    }
}
