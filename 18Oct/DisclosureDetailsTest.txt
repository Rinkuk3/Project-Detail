import com.td.mbna.msec.midtier.types.common.DisclosureInfoType;
import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class DisclosureDetailsTest {

    @Test
    public void testGettersAndSetters() {
        // Given
        DisclosureDetails disclosureDetails = new DisclosureDetails();
        DisclosureInfoType disclosureInfo = new DisclosureInfoType(); // Assuming this is a valid object
        String disclosureText = "This is a sample disclosure text.";

        // When
        disclosureDetails.setDisclosureInfo(disclosureInfo);
        disclosureDetails.setDisclosureText(disclosureText);

        // Then
        assertThat(disclosureDetails.getDisclosureInfo()).isEqualTo(disclosureInfo);
        assertThat(disclosureDetails.getDisclosureText()).isEqualTo(disclosureText);
    }

    @Test
    public void testSetDisclosureInfo() {
        // Given
        DisclosureDetails disclosureDetails = new DisclosureDetails();
        DisclosureInfoType disclosureInfo = new DisclosureInfoType();

        // When
        disclosureDetails.setDisclosureInfo(disclosureInfo);

        // Then
        assertThat(disclosureDetails.getDisclosureInfo()).isEqualTo(disclosureInfo);
    }

    @Test
    public void testSetDisclosureText() {
        // Given
        DisclosureDetails disclosureDetails = new DisclosureDetails();
        String disclosureText = "Sample disclosure text.";

        // When
        disclosureDetails.setDisclosureText(disclosureText);

        // Then
        assertThat(disclosureDetails.getDisclosureText()).isEqualTo(disclosureText);
    }

    @Test
    public void testDefaultConstructor() {
        // Given
        DisclosureDetails disclosureDetails = new DisclosureDetails();

        // When & Then
        assertThat(disclosureDetails.getDisclosureInfo()).isNull();
        assertThat(disclosureDetails.getDisclosureText()).isNull();
    }
}
