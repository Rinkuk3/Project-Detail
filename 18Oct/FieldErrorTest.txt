import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class FieldErrorTest {

    @Test
    public void testGettersAndSetters() {
        // Given
        FieldError fieldError = new FieldError();
        String errorCode = "400";
        String errorDescription = "Bad Request";
        String field = "email";

        // When
        fieldError.setErrorCode(errorCode);
        fieldError.setErrorDescription(errorDescription);
        fieldError.setField(field);

        // Then
        assertThat(fieldError.getErrorCode()).isEqualTo(errorCode);
        assertThat(fieldError.getErrorDescription()).isEqualTo(errorDescription);
        assertThat(fieldError.getField()).isEqualTo(field);
    }

    @Test
    public void testSetErrorCode() {
        // Given
        FieldError fieldError = new FieldError();
        String errorCode = "404";

        // When
        fieldError.setErrorCode(errorCode);

        // Then
        assertThat(fieldError.getErrorCode()).isEqualTo(errorCode);
    }

    @Test
    public void testSetErrorDescription() {
        // Given
        FieldError fieldError = new FieldError();
        String errorDescription = "Not Found";

        // When
        fieldError.setErrorDescription(errorDescription);

        // Then
        assertThat(fieldError.getErrorDescription()).isEqualTo(errorDescription);
    }

    @Test
    public void testSetField() {
        // Given
        FieldError fieldError = new FieldError();
        String field = "username";

        // When
        fieldError.setField(field);

        // Then
        assertThat(fieldError.getField()).isEqualTo(field);
    }

    @Test
    public void testDefaultConstructor() {
        // Given
        FieldError fieldError = new FieldError();

        // When & Then
        assertThat(fieldError.getErrorCode()).isNull();
        assertThat(fieldError.getErrorDescription()).isNull();
        assertThat(fieldError.getField()).isNull();
    }
}
