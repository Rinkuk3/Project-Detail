import org.junit.jupiter.api.Test;

import static org.assertj.core.api.Assertions.assertThat;

public class AccountAlreadyExistExceptionTest {

    @Test
    public void testAccountAlreadyExistExceptionMessage() {
        // Given
        String accountNumber = "123456789";
        
        // When
        AccountAlreadyExistException exception = new AccountAlreadyExistException(accountNumber);

        // Then
        assertThat(exception.getMessage()).isEqualTo("Invalid account: Account with number " + accountNumber + " does already exist!");
    }

    @Test
    public void testAccountAlreadyExistExceptionSerialization() {
        // Given
        String accountNumber = "987654321";
        AccountAlreadyExistException exception = new AccountAlreadyExistException(accountNumber);

        // When
        String expectedMessage = "Invalid account: Account with number " + accountNumber + " does already exist!";
        
        // Then
        assertThat(exception.getMessage()).isEqualTo(expectedMessage);
        assertThat(exception).isInstanceOf(MBNAException.class);
    }
}
