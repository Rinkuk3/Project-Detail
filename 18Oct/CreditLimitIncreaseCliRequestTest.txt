package com.td.mbna.msec.creditlimit.model;

import static org.junit.jupiter.api.Assertions.*;
import java.math.BigDecimal;

import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import com.td.mbna.msec.customer.domain.EmployerInfo;
import com.td.mbna.msec.customer.domain.OtherIncome;

public class CreditLimitIncreaseCliRequestTest {

    private CreditLimitIncreaseCliRequest request;

    @BeforeEach
    public void setUp() {
        request = new CreditLimitIncreaseCliRequest();
    }

    @Test
    public void testSetAndGetRequestCreditLimitIncrease() {
        BigDecimal creditLimitIncrease = BigDecimal.valueOf(5000);
        request.setRequestCreditLimitIncrease(creditLimitIncrease);
        assertEquals(creditLimitIncrease, request.getRequestCreditLimitIncrease());
    }

    @Test
    public void testSetAndGetEmployerInfo() {
        EmployerInfo employerInfo = new EmployerInfo();
        request.setEmployerInfo(employerInfo);
        assertEquals(employerInfo, request.getEmployerInfo());
    }

    @Test
    public void testSetAndGetHousingStatus() {
        String housingStatus = "Owned";
        request.setHousingStatus(housingStatus);
        assertEquals(housingStatus, request.getHousingStatus());
    }

    @Test
    public void testSetAndGetMonthlyHousingPayments() {
        BigDecimal monthlyPayments = BigDecimal.valueOf(1200);
        request.setMonthlyHousingPayments(monthlyPayments);
        assertEquals(monthlyPayments, request.getMonthlyHousingPayments());
    }

    @Test
    public void testSetAndGetOtherIncome() {
        OtherIncome otherIncome = new OtherIncome();
        request.setOtherIncome(otherIncome);
        assertEquals(otherIncome, request.getOtherIncome());
    }

    @Test
    public void testSetAndGetLimitConsent() {
        request.setLimitConsent(true);
        assertTrue(request.getLimitConsent());
        request.setLimitConsent(false);
        assertFalse(request.getLimitConsent());
    }
}
