import static org.junit.jupiter.api.Assertions.*;
import java.math.BigDecimal;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class AddAuResponseTest {

    private AddAuResponse addAuResponse;

    @BeforeEach
    public void setUp() {
        addAuResponse = new AddAuResponse();
    }

    @Test
    public void testStatus() {
        String status = "SUCCESS";
        addAuResponse.setStatus(status);
        assertEquals(status, addAuResponse.getStatus());
    }

    @Test
    public void testErrorCode() {
        String errorCode = "ERR001";
        addAuResponse.setErrorCode(errorCode);
        assertEquals(errorCode, addAuResponse.getErrorCode());
    }

    @Test
    public void testAuFee() {
        BigDecimal auFee = BigDecimal.valueOf(100.00);
        addAuResponse.setAuFee(auFee);
        assertEquals(auFee, addAuResponse.getAuFee());
    }

    @Test
    public void testAnnualFee() {
        BigDecimal annualFee = BigDecimal.valueOf(200.00);
        addAuResponse.setAnnualFee(annualFee);
        assertEquals(annualFee, addAuResponse.getAnnualFee());
    }

    @Test
    public void testVerificationPageMessage() {
        String message = "Verification page message";
        addAuResponse.setVerificationPageMessage(message);
        assertEquals(message, addAuResponse.getVerificationPageMessage());
    }

    @Test
    public void testConfirmationPageMessage() {
        String message = "Confirmation page message";
        addAuResponse.setConfirmationPageMessage(message);
        assertEquals(message, addAuResponse.getConfirmationPageMessage());
    }
}
