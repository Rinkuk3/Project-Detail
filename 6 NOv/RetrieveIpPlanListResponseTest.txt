package com.td.mbna.msec.midtier.account.payload.retrieveipplanlist;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertNull;

import com.td.mbna.msec.midtier.types.common.BaseResponse;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

public class RetrieveIpPlanListResponseTest {

    private RetrieveIpPlanListResponse retrieveIpPlanListResponse;
    private ResponsePayload mockResponsePayload;

    @BeforeEach
    public void setUp() {
        retrieveIpPlanListResponse = new RetrieveIpPlanListResponse();
        mockResponsePayload = new ResponsePayload(); // Assuming ResponsePayload is a simple POJO
    }

    @Test
    public void testGetResponsePayload_DefaultValue() {
        assertNull(retrieveIpPlanListResponse.getResponsePayload(), 
                   "Default value of responsePayload should be null");
    }

    @Test
    public void testSetAndGetResponsePayload() {
        retrieveIpPlanListResponse.setResponsePayload(mockResponsePayload);
        assertEquals(mockResponsePayload, retrieveIpPlanListResponse.getResponsePayload(),
                     "getResponsePayload should return the set ResponsePayload object");
    }

    @Test
    public void testBaseResponseInheritance() {
        assertEquals(BaseResponse.class, retrieveIpPlanListResponse.getClass().getSuperclass(),
                     "RetrieveIpPlanListResponse should inherit from BaseResponse");
    }
}
