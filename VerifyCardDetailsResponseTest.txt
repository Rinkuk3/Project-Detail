import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class VerifyCardDetailsResponseTest {

    private VerifyCardDetailsResponse verifyCardDetailsResponse;

    @BeforeEach
    void setUp() {
        verifyCardDetailsResponse = new VerifyCardDetailsResponse();
    }

    @Test
    void testDefaultValues() {
        // Verify that the default values are null for Boolean and String fields
        assertNull(verifyCardDetailsResponse.getSuccess());
        assertNull(verifyCardDetailsResponse.getErrorCode());
    }

    @Test
    void testSetterAndGetterForSuccess() {
        // Test the 'success' Boolean field
        verifyCardDetailsResponse.setSuccess(true);
        assertTrue(verifyCardDetailsResponse.getSuccess());

        verifyCardDetailsResponse.setSuccess(false);
        assertFalse(verifyCardDetailsResponse.getSuccess());

        verifyCardDetailsResponse.setSuccess(null);
        assertNull(verifyCardDetailsResponse.getSuccess());
    }

    @Test
    void testSetterAndGetterForErrorCode() {
        // Test the 'errorCode' String field
        verifyCardDetailsResponse.setErrorCode("ERROR_001");
        assertEquals("ERROR_001", verifyCardDetailsResponse.getErrorCode());

        verifyCardDetailsResponse.setErrorCode(null);
        assertNull(verifyCardDetailsResponse.getErrorCode());
    }

    @Test
    void testToString() {
        // Test toString method (generated by Lombok)
        verifyCardDetailsResponse.setSuccess(true);
        verifyCardDetailsResponse.setErrorCode("ERROR_001");

        String expectedString = "VerifyCardDetailsResponse(success=true, errorCode=ERROR_001)";
        assertEquals(expectedString, verifyCardDetailsResponse.toString());
    }
}
