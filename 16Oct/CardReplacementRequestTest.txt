import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class CardReplacementRequestTest {

    private CardReplacementRequest cardReplacementRequest;

    @BeforeEach
    public void setUp() {
        // Initialize the CardReplacementRequest object before each test
        cardReplacementRequest = new CardReplacementRequest();
    }

    @Test
    public void testGettersAndSetters() {
        // Arrange
        String aliasCustID = "12345";
        String reasonForRequest = "Lost Card";

        // Act
        cardReplacementRequest.setAliasCustID(aliasCustID);
        cardReplacementRequest.setReasonForRequest(reasonForRequest);

        // Assert
        assertEquals(aliasCustID, cardReplacementRequest.getAliasCustID(), "AliasCustID should match");
        assertEquals(reasonForRequest, cardReplacementRequest.getReasonForRequest(), "ReasonForRequest should match");
    }

    @Test
    public void testDefaultValues() {
        // Assert that the default values are null
        assertNull(cardReplacementRequest.getAliasCustID(), "AliasCustID should be null by default");
        assertNull(cardReplacementRequest.getReasonForRequest(), "ReasonForRequest should be null by default");
    }
}
