package com.td.mbna.msec.balancetransfer.domain;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class BalanceTransferToCreditorPayeeRequestTest {

    @Test
    void testSettersAndGetters() {
        // Create an instance of the class
        BalanceTransferToCreditorPayeeRequest request = new BalanceTransferToCreditorPayeeRequest();

        // Test id
        String id = "12345";
        request.setId(id);
        assertEquals(id, request.getId());

        // Test nickname
        String nickname = "My Payee";
        request.setNickname(nickname);
        assertEquals(nickname, request.getNickname());

        // Test oldPayeeNickname
        String oldNickname = "Old Payee";
        request.setOldPayeeNickname(oldNickname);
        assertEquals(oldNickname, request.getOldPayeeNickname());

        // Test institutionName
        String institutionName = "BMO";
        request.setInstitutionName(institutionName);
        assertEquals(institutionName, request.getInstitutionName());

        // Test merchantId
        String merchantId = "merchant_001";
        request.setMerchantId(merchantId);
        assertEquals(merchantId, request.getMerchantId());

        // Test accountId
        String accountId = "account_001";
        request.setAccountId(accountId);
        assertEquals(accountId, request.getAccountId());
    }
}
