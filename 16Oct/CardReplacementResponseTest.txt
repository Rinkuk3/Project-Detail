import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class CardReplacementResponseTest {

    @Test
    public void testBuilder() {
        // Arrange
        String aliasCustID = "12345";
        String reasonForRequest = "Lost Card";

        // Act
        CardReplacementResponse response = CardReplacementResponse.builder()
                .aliasCustID(aliasCustID)
                .reasonForRequest(reasonForRequest)
                .build();

        // Assert
        assertEquals(aliasCustID, response.getAliasCustID(), "AliasCustID should match");
        assertEquals(reasonForRequest, response.getReasonForRequest(), "ReasonForRequest should match");
    }

    @Test
    public void testGetters() {
        // Arrange
        String aliasCustID = "67890";
        String reasonForRequest = "Stolen Card";
        CardReplacementResponse response = new CardReplacementResponse();
        response.setAliasCustID(aliasCustID);
        response.setReasonForRequest(reasonForRequest);

        // Act & Assert
        assertEquals(aliasCustID, response.getAliasCustID(), "AliasCustID should match");
        assertEquals(reasonForRequest, response.getReasonForRequest(), "ReasonForRequest should match");
    }

    @Test
    public void testDefaultValues() {
        // Act
        CardReplacementResponse response = new CardReplacementResponse();

        // Assert
        assertNull(response.getAliasCustID(), "AliasCustID should be null by default");
        assertNull(response.getReasonForRequest(), "ReasonForRequest should be null by default");
    }
}
