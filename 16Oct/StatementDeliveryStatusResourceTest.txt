import com.td.mbna.msec.account.domain.StatementDeliveryStatusRequest;
import com.td.mbna.msec.account.statement.StatementDeliveryStatusService;
import com.td.mbna.msec.common.domain.BaseMsecbffResponse;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.MockitoAnnotations;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;

import javax.servlet.http.HttpServletRequest;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.mockito.Mockito.when;

public class StatementDeliveryStatusResourceTest {

    @InjectMocks
    private StatementDeliveryStatusResource statementDeliveryStatusResource;

    @Mock
    private StatementDeliveryStatusService statementDeliveryStatusService;

    @Mock
    private HttpServletRequest httpServletRequest;

    @Mock
    private StatementDeliveryStatusRequest statementDeliveryStatusRequest;

    @BeforeEach
    public void setUp() {
        MockitoAnnotations.openMocks(this);
    }

    @Test
    public void testUpdateStatementDeliveryStatus_Success() {
        // Arrange
        String accountNumber = "123456";
        BaseMsecbffResponse expectedResponse = new BaseMsecbffResponse();
        expectedResponse.setSuccess(true);
        when(statementDeliveryStatusRequest.getAccountNumber()).thenReturn(accountNumber);
        when(statementDeliveryStatusService.updateDeliveryPreferencesToPaper(accountNumber, httpServletRequest)).thenReturn(expectedResponse);

        // Act
        BaseMsecbffResponse actualResponse = statementDeliveryStatusResource.updateStatementDeliveryStatus(statementDeliveryStatusRequest, httpServletRequest);

        // Assert
        assertEquals(expectedResponse, actualResponse);
    }

    @Test
    public void testUpdateStatementDeliveryStatus_Failure() {
        // Arrange
        String accountNumber = "654321";
        BaseMsecbffResponse expectedResponse = new BaseMsecbffResponse();
        expectedResponse.setSuccess(false);
        when(statementDeliveryStatusRequest.getAccountNumber()).thenReturn(accountNumber);
        when(statementDeliveryStatusService.updateDeliveryPreferencesToPaper(accountNumber, httpServletRequest)).thenReturn(expectedResponse);

        // Act
        BaseMsecbffResponse actualResponse = statementDeliveryStatusResource.updateStatementDeliveryStatus(statementDeliveryStatusRequest, httpServletRequest);

        // Assert
        assertEquals(expectedResponse, actualResponse);
    }
}
