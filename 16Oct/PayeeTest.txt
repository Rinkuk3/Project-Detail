package com.td.mbna.msec.balancetransfer.domain;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class PayeeTest {

    @Test
    void testSettersAndGetters() {
        // Arrange
        Payee payee = new Payee();
        String testId = "123";
        String testNickname = "My Payee";
        String testInstitutionName = "Bank of America";
        String testMerchantId = "456";
        String testAccountId = "78901234";

        // Act
        payee.setId(testId);
        payee.setNickname(testNickname);
        payee.setInstitutionName(testInstitutionName);
        payee.setMerchantId(testMerchantId);
        payee.setAccountId(testAccountId);

        // Assert
        assertEquals(testId, payee.getId());
        assertEquals(testNickname, payee.getNickname());
        assertEquals(testInstitutionName, payee.getInstitutionName());
        assertEquals(testMerchantId, payee.getMerchantId());
        assertEquals(testAccountId, payee.getAccountId());
    }

    @Test
    void testGetShortAccountId() {
        // Arrange
        Payee payee = new Payee();
        
        // Test with a valid account ID
        String testAccountId = "78901234";
        payee.setAccountId(testAccountId);
        
        // Act
        String shortAccountId = payee.getShortAccountId();
        
        // Assert
        assertEquals("1234", shortAccountId);

        // Test with a short account ID
        payee.setAccountId("12");
        assertEquals("12", payee.getShortAccountId());

        // Test with null account ID
        payee.setAccountId(null);
        assertNull(payee.getShortAccountId());
    }

    @Test
    void testToString() {
        // Arrange
        Payee payee = new Payee();
        payee.setId("123");
        payee.setNickname("My Payee");
        payee.setInstitutionName("Bank of America");
        payee.setMerchantId("456");
        payee.setAccountId("78901234");

        // Act
        String result = payee.toString();

        // Assert
        String expected = "id=123,nn=My Payee,cci=Bank of America,merchant 456,acct=78901234";
        assertEquals(expected, result);
    }

    @Test
    void testPayeeTypeEnum() {
        // Test the enum values
        assertEquals("CHEQUING", Payee.Type.CHEQUING.name());
        assertEquals("CREDIT_CARD", Payee.Type.CREDIT_CARD.name());

        // Test the enum ordinal values
        assertEquals(0, Payee.Type.CHEQUING.ordinal());
        assertEquals(1, Payee.Type.CREDIT_CARD.ordinal());
    }
}
