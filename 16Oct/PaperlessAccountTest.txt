import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class PaperlessAccountTest {

    @Test
    public void testPaperlessAccountBuilder() {
        // Arrange
        String expectedCardEndingIn = "1234";
        String expectedAccountId = "acc-001";
        String expectedEmail = "test@example.com";
        String expectedCardName = "Gold Card";
        String expectedFrCardName = "Carte Or";
        String expectedCardArtUrl = "http://example.com/cardArt.png";
        String expectedFrCardArtUrl = "http://example.com/frCardArt.png";
        boolean expectedHybrid = true;

        // Act
        PaperlessAccount account = PaperlessAccount.builder()
                .cardEndingIn(expectedCardEndingIn)
                .accountId(expectedAccountId)
                .email(expectedEmail)
                .cardName(expectedCardName)
                .frCardName(expectedFrCardName)
                .cardArtUrl(expectedCardArtUrl)
                .frCardArtUrl(expectedFrCardArtUrl)
                .hybrid(expectedHybrid)
                .build();

        // Assert
        assertEquals(expectedCardEndingIn, account.getCardEndingIn());
        assertEquals(expectedAccountId, account.getAccountId());
        assertEquals(expectedEmail, account.getEmail());
        assertEquals(expectedCardName, account.getCardName());
        assertEquals(expectedFrCardName, account.getFrCardName());
        assertEquals(expectedCardArtUrl, account.getCardArtUrl());
        assertEquals(expectedFrCardArtUrl, account.getFrCardArtUrl());
        assertEquals(expectedHybrid, account.isHybrid());
    }

    @Test
    public void testDefaultValues() {
        // Act
        PaperlessAccount account = new PaperlessAccount();

        // Assert
        assertNull(account.getCardEndingIn());
        assertNull(account.getAccountId());
        assertNull(account.getEmail());
        assertNull(account.getCardName());
        assertNull(account.getFrCardName());
        assertNull(account.getCardArtUrl());
        assertNull(account.getFrCardArtUrl());
        assertFalse(account.isHybrid());
    }

    @Test
    public void testSetters() {
        // Arrange
        PaperlessAccount account = new PaperlessAccount();
        String cardEndingIn = "5678";
        String accountId = "acc-002";
        String email = "test2@example.com";
        String cardName = "Platinum Card";
        String frCardName = "Carte Platine";
        String cardArtUrl = "http://example.com/platinumCardArt.png";
        String frCardArtUrl = "http://example.com/frPlatinumCardArt.png";
        boolean hybrid = false;

        // Act
        account.setCardEndingIn(cardEndingIn);
        account.setAccountId(accountId);
        account.setEmail(email);
        account.setCardName(cardName);
        account.setFrCardName(frCardName);
        account.setCardArtUrl(cardArtUrl);
        account.setFrCardArtUrl(frCardArtUrl);
        account.setHybrid(hybrid);

        // Assert
        assertEquals(cardEndingIn, account.getCardEndingIn());
        assertEquals(accountId, account.getAccountId());
        assertEquals(email, account.getEmail());
        assertEquals(cardName, account.getCardName());
        assertEquals(frCardName, account.getFrCardName());
        assertEquals(cardArtUrl, account.getCardArtUrl());
        assertEquals(frCardArtUrl, account.getFrCardArtUrl());
        assertEquals(hybrid, account.isHybrid());
    }
}
