package com.td.mbna.msec.auditlogging.model;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class LoggingResponseTest {

    @Test
    void testConstructor() {
        String result = "Success";
        String message = "Log created successfully";

        LoggingResponse response = new LoggingResponse(result, message);
        
        assertEquals(result, response.getResult());
        assertEquals(message, response.getMessage());
    }

    @Test
    void testSetAndGetResult() {
        LoggingResponse response = new LoggingResponse("InitialResult", "InitialMessage");
        
        String newResult = "UpdatedResult";
        response.setResult(newResult);
        
        assertEquals(newResult, response.getResult());
    }

    @Test
    void testSetAndGetMessage() {
        LoggingResponse response = new LoggingResponse("InitialResult", "InitialMessage");
        
        String newMessage = "UpdatedMessage";
        response.setMessage(newMessage);
        
        assertEquals(newMessage, response.getMessage());
    }
}
