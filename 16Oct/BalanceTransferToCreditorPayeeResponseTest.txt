package com.td.mbna.msec.balancetransfer.domain;

import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class BalanceTransferToCreditorPayeeResponseTest {

    @Test
    void testBuilderAndGetters() {
        // Arrange
        String id = "12345";
        String nickname = "My Payee";
        String institutionName = "BMO";
        String merchantId = "merchant_001";
        String accountId = "account_001";

        // Act
        BalanceTransferToCreditorPayeeResponse response = BalanceTransferToCreditorPayeeResponse.builder()
                .id(id)
                .nickname(nickname)
                .institutionName(institutionName)
                .merchantId(merchantId)
                .accountId(accountId)
                .build();

        // Assert
        assertEquals(id, response.getId());
        assertEquals(nickname, response.getNickname());
        assertEquals(institutionName, response.getInstitutionName());
        assertEquals(merchantId, response.getMerchantId());
        assertEquals(accountId, response.getAccountId());
    }

    @Test
    void testDefaultValues() {
        // Act
        BalanceTransferToCreditorPayeeResponse response = BalanceTransferToCreditorPayeeResponse.builder().build();

        // Assert
        assertNull(response.getId());
        assertNull(response.getNickname());
        assertNull(response.getInstitutionName());
        assertNull(response.getMerchantId());
        assertNull(response.getAccountId());
    }
}
