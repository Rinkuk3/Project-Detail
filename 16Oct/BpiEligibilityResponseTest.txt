import com.td.mbna.msec.account.domain.BpiEligibilityResponse;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class BpiEligibilityResponseTest {

    @Test
    public void testDefaultConstructorAndGetters() {
        // Arrange
        BpiEligibilityResponse response = new BpiEligibilityResponse();

        // Act & Assert
        assertFalse(response.isEligible(), "Eligible should default to false");
        assertFalse(response.isAccepted(), "Accepted should default to false");
    }

    @Test
    public void testSettersAndGetters() {
        // Arrange
        BpiEligibilityResponse response = new BpiEligibilityResponse();

        // Act
        response.setEligible(true);
        response.setAccepted(false);

        // Assert
        assertTrue(response.isEligible(), "Eligible should be set to true");
        assertFalse(response.isAccepted(), "Accepted should be set to false");

        // Act with other values
        response.setEligible(false);
        response.setAccepted(true);

        // Assert again
        assertFalse(response.isEligible(), "Eligible should be set to false");
        assertTrue(response.isAccepted(), "Accepted should be set to true");
    }
}
