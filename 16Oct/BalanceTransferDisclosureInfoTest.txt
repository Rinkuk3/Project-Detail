package com.td.mbna.msec.balancetransfer.domain;

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

class BalanceTransferDisclosureInfoTest {

    @Test
    void testBuilder() {
        String expectedText = "This is an offer disclosure text.";
        
        // Create an instance using the builder
        BalanceTransferDisclosureInfo info = BalanceTransferDisclosureInfo.builder()
                .offerDisclosureText(expectedText)
                .build();
        
        // Assert that the offerDisclosureText is set correctly
        assertEquals(expectedText, info.getOfferDisclosureText());
    }

    @Test
    void testSetAndGetOfferDisclosureText() {
        BalanceTransferDisclosureInfo info = new BalanceTransferDisclosureInfo();
        
        String expectedText = "Updated offer disclosure text.";
        info.setOfferDisclosureText(expectedText);
        
        // Assert that the offerDisclosureText is set and retrieved correctly
        assertEquals(expectedText, info.getOfferDisclosureText());
    }
}
