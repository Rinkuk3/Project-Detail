import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class UrpTest {

    @Test
    public void testGettersAndSetters() {
        // Arrange
        Urp urp = new Urp();
        boolean newlyOpenedAccount = true;
        boolean newlyOpenedAccountBT = false;
        boolean newlyOpenedAccountTTC = true;
        boolean newlyOpenedCustomer = false;

        // Act
        urp.setNewlyOpenedAccount(newlyOpenedAccount);
        urp.setNewlyOpenedAccountBT(newlyOpenedAccountBT);
        urp.setNewlyOpenedAccountTTC(newlyOpenedAccountTTC);
        urp.setNewlyOpenedCustomer(newlyOpenedCustomer);

        // Assert
        assertTrue(urp.isNewlyOpenedAccount());
        assertFalse(urp.isNewlyOpenedAccountBT());
        assertTrue(urp.isNewlyOpenedAccountTTC());
        assertFalse(urp.isNewlyOpenedCustomer());
    }

    @Test
    public void testDefaultValues() {
        // Arrange
        Urp urp = new Urp();

        // Act & Assert
        assertFalse(urp.isNewlyOpenedAccount()); // Default value should be false
        assertFalse(urp.isNewlyOpenedAccountBT()); // Default value should be false
        assertFalse(urp.isNewlyOpenedAccountTTC()); // Default value should be false
        assertFalse(urp.isNewlyOpenedCustomer()); // Default value should be false
    }
}
