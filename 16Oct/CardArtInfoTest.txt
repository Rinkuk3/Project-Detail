import com.td.mbna.msec.account.domain.CardArtInfo;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class CardArtInfoTest {

    @Test
    public void testDefaultConstructor() {
        // Arrange
        CardArtInfo cardArtInfo = new CardArtInfo();

        // Assert
        assertNull(cardArtInfo.getLanguage(), "Language should be null");
        assertNull(cardArtInfo.getCardName(), "Card name should be null");
        assertNull(cardArtInfo.getCardNameShort(), "Card name short should be null");
        assertNull(cardArtInfo.getCardUrl(), "Card URL should be null");
        assertNull(cardArtInfo.getAffinityLogoUrl(), "Affinity logo URL should be null");
    }

    @Test
    public void testParameterizedConstructor() {
        // Arrange
        String language = "en";
        String cardName = "Visa Card";
        String cardNameShort = "Visa";
        String cardUrl = "http://example.com/card.png";
        String affinityLogoUrl = "http://example.com/logo.png";

        // Act
        CardArtInfo cardArtInfo = new CardArtInfo(language, cardName, cardNameShort, cardUrl, affinityLogoUrl);

        // Assert
        assertEquals(language, cardArtInfo.getLanguage(), "Language should match");
        assertEquals(cardName, cardArtInfo.getCardName(), "Card name should match");
        assertEquals(cardNameShort, cardArtInfo.getCardNameShort(), "Card name short should match");
        assertEquals(cardUrl, cardArtInfo.getCardUrl(), "Card URL should match");
        assertEquals(affinityLogoUrl, cardArtInfo.getAffinityLogoUrl(), "Affinity logo URL should match");
    }

    @Test
    public void testSetters() {
        // Arrange
        CardArtInfo cardArtInfo = new CardArtInfo();

        // Act
        cardArtInfo.setLanguage("fr");
        cardArtInfo.setCardName("Carte Visa");
        cardArtInfo.setCardNameShort("Visa");
        cardArtInfo.setCardUrl("http://example.com/fr_card.png");
        cardArtInfo.setAffinityLogoUrl("http://example.com/fr_logo.png");

        // Assert
        assertEquals("fr", cardArtInfo.getLanguage(), "Language should be updated to 'fr'");
        assertEquals("Carte Visa", cardArtInfo.getCardName(), "Card name should be updated to 'Carte Visa'");
        assertEquals("Visa", cardArtInfo.getCardNameShort(), "Card name short should be updated to 'Visa'");
        assertEquals("http://example.com/fr_card.png", cardArtInfo.getCardUrl(), "Card URL should be updated");
        assertEquals("http://example.com/fr_logo.png", cardArtInfo.getAffinityLogoUrl(), "Affinity logo URL should be updated");
    }

    @Test
    public void testMethodChainingForSetters() {
        // Arrange
        CardArtInfo cardArtInfo = new CardArtInfo();

        // Act
        CardArtInfo returnedObject = cardArtInfo.setCardUrl("http://example.com/card.png")
                                                 .setAffinityLogoUrl("http://example.com/logo.png");

        // Assert
        assertSame(cardArtInfo, returnedObject, "Method chaining should return the same instance");
        assertEquals("http://example.com/card.png", cardArtInfo.getCardUrl(), "Card URL should be set correctly");
        assertEquals("http://example.com/logo.png", cardArtInfo.getAffinityLogoUrl(), "Affinity logo URL should be set correctly");
    }
}
