import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class IPPlanDetailsRequestTest {

    @Test
    public void testDefaultConstructor() {
        // Arrange
        IPPlanDetailsRequest request = new IPPlanDetailsRequest();

        // Act
        // No action needed; just check default values

        // Assert
        assertNull(request.getId(), "Default id should be null");
        assertNull(request.getPlanName(), "Default planName should be null");
        assertNull(request.getAccountNumber(), "Default accountNumber should be null");
    }

    @Test
    public void testSettersAndGetters() {
        // Arrange
        IPPlanDetailsRequest request = new IPPlanDetailsRequest();
        String testId = "123";
        String testPlanName = "Premium Plan";
        String testAccountNumber = "ACC-456";

        // Act
        request.setId(testId);
        request.setPlanName(testPlanName);
        request.setAccountNumber(testAccountNumber);

        // Assert
        assertEquals(testId, request.getId(), "The id should match the set value");
        assertEquals(testPlanName, request.getPlanName(), "The planName should match the set value");
        assertEquals(testAccountNumber, request.getAccountNumber(), "The accountNumber should match the set value");
    }
}
