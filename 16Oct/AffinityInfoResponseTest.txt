import com.td.mbna.msec.account.domain.AffinityInfoResponse;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

public class AffinityInfoResponseTest {

    @Test
    public void testGettersAndSetters() {
        // Arrange
        AffinityInfoResponse response = AffinityInfoResponse.builder()
                .productCode("PC123")
                .language("EN")
                .cardName("Visa Platinum")
                .build();
        
        // Act & Assert
        assertEquals("PC123", response.getProductCode(), "Product code should match");
        assertEquals("EN", response.getLanguage(), "Language should match");
        assertEquals("Visa Platinum", response.getCardName(), "Card name should match");
    }

    @Test
    public void testBuilder() {
        // Arrange & Act
        AffinityInfoResponse response = AffinityInfoResponse.builder()
                .productCode("PC456")
                .language("FR")
                .cardName("MasterCard Gold")
                .build();
        
        // Assert
        assertNotNull(response, "AffinityInfoResponse object should not be null");
        assertEquals("PC456", response.getProductCode(), "Product code should match");
        assertEquals("FR", response.getLanguage(), "Language should match");
        assertEquals("MasterCard Gold", response.getCardName(), "Card name should match");
    }

    @Test
    public void testDefaultConstructor() {
        // Act
        AffinityInfoResponse response = new AffinityInfoResponse();
        
        // Assert
        assertNull(response.getProductCode(), "Product code should be null");
        assertNull(response.getLanguage(), "Language should be null");
        assertNull(response.getCardName(), "Card name should be null");
    }
}
