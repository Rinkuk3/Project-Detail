import com.td.mbna.msec.account.domain.mobile.OfferAttributes;
import com.td.mbna.msec.account.domain.mobile.Fees;
import com.td.mbna.msec.account.domain.mobile.Duration;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;

import static org.junit.jupiter.api.Assertions.*;

class OfferAttributesTest {

    private OfferAttributes offerAttributes;
    private Fees fees;
    private Duration duration;

    @BeforeEach
    void setUp() {
        fees = new Fees(100, 2.5);
        duration = new Duration(12, 6, "2024-12-31");
        offerAttributes = new OfferAttributes("Terms and conditions", 3.5, fees, duration);
    }

    @Test
    void testConstructorAndGetters() {
        // Act & Assert
        assertEquals("Terms and conditions", offerAttributes.getTerms());
        assertEquals(3.5, offerAttributes.getAir());
        assertEquals(fees, offerAttributes.getFees());
        assertEquals(duration, offerAttributes.getDuration());
    }

    @Test
    void testSetTerms() {
        // Arrange
        String newTerms = "Updated terms and conditions";
        
        // Act
        offerAttributes.setTerms(newTerms);
        
        // Assert
        assertEquals(newTerms, offerAttributes.getTerms());
    }

    @Test
    void testSetAir() {
        // Arrange
        Double newAir = 4.0;
        
        // Act
        offerAttributes.setAir(newAir);
        
        // Assert
        assertEquals(newAir, offerAttributes.getAir());
    }

    @Test
    void testSetFees() {
        // Arrange
        Fees newFees = new Fees(200, 3.0);
        
        // Act
        offerAttributes.setFees(newFees);
        
        // Assert
        assertEquals(newFees, offerAttributes.getFees());
    }

    @Test
    void testSetDuration() {
        // Arrange
        Duration newDuration = new Duration(24, 12, "2025-06-30");
        
        // Act
        offerAttributes.setDuration(newDuration);
        
        // Assert
        assertEquals(newDuration, offerAttributes.getDuration());
    }

}
